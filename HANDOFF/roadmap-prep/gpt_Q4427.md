# Executive verdict

The primary-decomposition argument is correct and genuinely independent of Mordell–Weil finite generation.

The clean implementation is:

1. prove, purely in AddCommGroup, that a group killed by 21 is additively equivalent to its 3-primary and 7-primary subgroups;

1. identify the 3-primary subgroup with G / 3G by an explicit first-isomorphism-theorem map;

1. inject the 7-primary subgroup into the seven-element good-reduction group;

1. use explicit points of orders 3 and 7 to obtain a point of order 21, hence an equivalence ZMod 21 ≃+ E₀(L);

1. represent every degree-two fiber by two finite slots indexed by ZMod 21 × Fin 2. This produces at most 42 candidates without invoking Faltings, an abstract Jacobian torsion theorem, or Mordell–Weil finite generation.

Two points should be kept explicit in the integration:

- Finiteness and the upper bound #E₀(L) ≤ 21 do not by themselves prove equality. The order-21 point certificate is required.

variable {G : Type*} [AddCommGroup G]

/-- The subgroup killed by `n`. -/
def nTorsion (n : ℕ) : AddSubgroup G where
  carrier := {x | n • x = 0}
  zero_mem' := by simp
  add_mem' := by
    intro x y hx hy
    simpa [nsmul_add, hx, hy]
  neg_mem' := by
    intro x hx
    simpa [nsmul_neg, hx]

@[simp]
theorem mem_nTorsion {n : ℕ} {x : G} :
    x ∈ nTorsion (G := G) n ↔ n • x = 0 :=
  Iff.rfl

/-- Multiplication by `n` as an additive homomorphism. -/
def nsmulHom (n : ℕ) : G →+ G where
  toFun x := n • x
  map_zero' := by simp
  map_add' x y := by simp [nsmul_add]

@[simp]
theorem nsmulHom_apply (n : ℕ) (x : G) :
    nsmulHom (G := G) n x = n • x :=
  rfl

/-- The image `nG`. -/
def nRange (n : ℕ) : AddSubgroup G :=
  (nsmulHom (G := G) n).range

@[simp]
theorem mem_nRange {n : ℕ} {x : G} :
    x ∈ nRange (G := G) n ↔ ∃ y : G, n • y = x :=
  Iff.rfl

abbrev ThreePrimary (G : Type*) [AddCommGroup G] : AddSubgroup G :=
  nTorsion (G := G) 3

abbrev SevenPrimary (G : Type*) [AddCommGroup G] : AddSubgroup G :=
  nTorsion (G := G) 7

abbrev ThreeRange (G : Type*) [AddCommGroup G] : AddSubgroup G :=
  nRange (G := G) 3

abbrev SevenRange (G : Type*) [AddCommGroup G] : AddSubgroup G :=
  nRange (G := G) 7

lemma seven_nsmul_eq_self_of_three_nsmul_eq_zero
    {x : G} (hx : (3 : ℕ) • x = 0) :
    (7 : ℕ) • x = x := by
  calc
    (7 : ℕ) • x = (1 + 2 * 3 : ℕ) • x := by norm_num
    _ = (1 : ℕ) • x + (2 * 3 : ℕ) • x := by rw [add_nsmul]
    _ = x + (2 : ℕ) • ((3 : ℕ) • x) := by rw [one_nsmul, mul_nsmul]
    _ = x := by simp [hx]

lemma fifteen_nsmul_eq_self_of_seven_nsmul_eq_zero
    {x : G} (hx : (7 : ℕ) • x = 0) :
    (15 : ℕ) • x = x := by
  calc
    (15 : ℕ) • x = (1 + 2 * 7 : ℕ) • x := by norm_num
    _ = (1 : ℕ) • x + (2 * 7 : ℕ) • x := by rw [add_nsmul]
    _ = x + (2 : ℕ) • ((7 : ℕ) • x) := by rw [one_nsmul, mul_nsmul]
    _ = x := by simp [hx]

lemma fifteen_nsmul_eq_zero_of_three_nsmul_eq_zero
    {x : G} (hx : (3 : ℕ) • x = 0) :
    (15 : ℕ) • x = 0 := by
  calc
    (15 : ℕ) • x = (5 : ℕ) • ((3 : ℕ) • x) := by
      rw [← mul_nsmul]
      norm_num
    _ = 0 := by simp [hx]

/--
For a group killed by `21`, the explicit projectors are

* `x ↦ 7x` onto the 3-primary part;
* `x ↦ 15x` onto the 7-primary part.

Their sum is `22x=x` because `21x=0`.
-/
noncomputable def primaryDecomposition
    (h21 : ∀ x : G, (21 : ℕ) • x = 0) :
    G ≃+ ThreePrimary G × SevenPrimary G where
  toFun x :=
    (⟨(7 : ℕ) • x, by
        change (3 : ℕ) • ((7 : ℕ) • x) = 0
        calc
          (3 : ℕ) • ((7 : ℕ) • x) = (21 : ℕ) • x := by
            rw [← mul_nsmul]
            norm_num
          _ = 0 := h21 x⟩,
     ⟨(15 : ℕ) • x, by
        change (7 : ℕ) • ((15 : ℕ) • x) = 0
        calc
          (7 : ℕ) • ((15 : ℕ) • x) = (5 : ℕ) • ((21 : ℕ) • x) := by
            simp only [← mul_nsmul]
            norm_num
          _ = 0 := by simp [h21 x]⟩)
  invFun p := (p.1 : G) + (p.2 : G)
  left_inv x := by
    change (7 : ℕ) • x + (15 : ℕ) • x = x
    calc
      (7 : ℕ) • x + (15 : ℕ) • x = (22 : ℕ) • x := by
        rw [← add_nsmul]
        norm_num
      _ = (1 + 21 : ℕ) • x := by norm_num
      _ = x := by rw [add_nsmul, one_nsmul, h21 x, add_zero]
  right_inv p := by
    rcases p with ⟨a, b⟩
    apply Prod.ext
    · apply Subtype.ext
      change (7 : ℕ) • ((a : G) + (b : G)) = (a : G)
      rw [nsmul_add,
        seven_nsmul_eq_self_of_three_nsmul_eq_zero a.property,
        b.property, add_zero]
    · apply Subtype.ext
      change (15 : ℕ) • ((a : G) + (b : G)) = (b : G)
      rw [nsmul_add,
        fifteen_nsmul_eq_zero_of_three_nsmul_eq_zero a.property,
        fifteen_nsmul_eq_self_of_seven_nsmul_eq_zero b.property,
        zero_add]
  map_add' x y := by
    apply Prod.ext
    · apply Subtype.ext
      simp [nsmul_add]
    · apply Subtype.ext
      simp [nsmul_add]

/-- Under `[21]G=0`, the image `7G` is exactly `G[3]`. -/
theorem sevenRange_eq_threePrimary
    (h21 : ∀ x : G, (21 : ℕ) • x = 0) :
    SevenRange G = ThreePrimary G := by
  ext x
  constructor
  · rintro ⟨y, rfl⟩
    change (3 : ℕ) • ((7 : ℕ) • y) = 0
    calc
      (3 : ℕ) • ((7 : ℕ) • y) = (21 : ℕ) • y := by
        rw [← mul_nsmul]
        norm_num
      _ = 0 := h21 y
  · intro hx
    refine ⟨x, ?_⟩
    exact seven_nsmul_eq_self_of_three_nsmul_eq_zero hx

/-- Under `[21]G=0`, the image `3G` is exactly `G[7]`. -/

  ext x
  constructor
  · rintro ⟨y, rfl⟩
    change (7 : ℕ) • ((3 : ℕ) • y) = 0

      (7 : ℕ) • ((3 : ℕ) • y) = (21 : ℕ) • y := by

      _ = 0 := h21 y
  · intro hx
    refine ⟨(5 : ℕ) • x, ?_⟩
    calc
      (3 : ℕ) • ((5 : ℕ) • x) = (15 : ℕ) • x := by
        rw [← mul_nsmul]
        norm_num
      _ = x := fifteen_nsmul_eq_self_of_seven_nsmul_eq_zero hx

/-- The 3-primary and 7-primary parts intersect trivially. -/
theorem threePrimary_inf_sevenPrimary_eq_bot :
    ThreePrimary G ⊓ SevenPrimary G = ⊥ := by
  apply le_antisymm
  · intro x hx
    rw [AddSubgroup.mem_bot]
    have h3 : (3 : ℕ) • (x : G) = 0 := hx.1
    have h7 : (7 : ℕ) • (x : G) = 0 := hx.2
    exact
      (seven_nsmul_eq_self_of_three_nsmul_eq_zero h3).symm.trans h7
  · exact bot_le

## Why this proves the 3-primary finiteness without finite generation

The finite 3-isogeny Selmer computation may export either of the following:

```javascript
instance : Finite (E0L.Point ⧸ N18Block7.ThreeRange E0L.Point)

theorem modThree_card_le :
  Nat.card
    (E0L.Point ⧸ N18Block7.ThreeRange E0L.Point) ≤ 3
```

or the stronger weak-descent statement

```javascript
∀ P : E0L.Point, ∃ T : H3, ∃ Q : E0L.Point,
  P = (T : E0L.Point) + 3 • Q.
```

The latter says that the three-element group H3 surjects onto E₀(L)/3E₀(L). Either formulation gives a finite quotient. No finite-generation theorem enters: a Kummer map is an injection of the quotient into a finite Selmer set before any rank formula is discussed.

Although G[3] is naturally an F₃-vector space, the equivalence

```javascript
G ⧸ ThreeRange G ≃+ ThreePrimary G
```

is stronger and avoids all module-instance plumbing.

namespace N18Block7

section CyclicTwentyOne

variable {G : Type*} [AddCommGroup G]

lemma addOrderOf_add_eq_twentyOne
    {T3 T7 : G}
    (h3 : addOrderOf T3 = 3)
    (h7 : addOrderOf T7 = 7) :
    addOrderOf (T3 + T7) = 21 := by
  simpa [h3, h7] using
    addOrderOf_add_eq_mul_addOrderOf_of_coprime
      (AddCommute.all T3 T7)
      (by norm_num : Nat.Coprime 3 7)

/-- The homomorphism `ZMod n → G`, `k ↦ k • T`. -/
noncomputable def zmodGeneratorHom
    {n : ℕ} (T : G) (hT : addOrderOf T = n) :
    ZMod n →+ G where
  toFun k := (ZMod.cast k : ℤ) • T
  map_zero' := by simp
  map_add' x y := by
    obtain ⟨a, rfl⟩ := ZMod.intCast_surjective x
    obtain ⟨b, rfl⟩ := ZMod.intCast_surjective y
    rw [← Int.cast_add]
    simp only [ZMod.coe_intCast, ← hT,
      mod_addOrderOf_zsmul, add_zsmul]

/-- Exact order of `T` makes the preceding homomorphism injective. -/
theorem zmodGeneratorHom_injective
    {n : ℕ} (T : G) (hT : addOrderOf T = n) :
    Function.Injective (zmodGeneratorHom T hT) := by
  rw [injective_iff_map_eq_zero]
  intro x hx
  obtain ⟨a, rfl⟩ := ZMod.intCast_surjective x
  have ha : a • T = 0 := by
    simpa [zmodGeneratorHom, ZMod.coe_intCast, ← hT,
      mod_addOrderOf_zsmul] using hx
  rw [ZMod.intCast_zmod_eq_zero_iff_dvd]
  rw [← hT]
  exact
    (addOrderOf_dvd_iff_zsmul_eq_zero
      (x := T) (i := a)).2 ha

/-- A group of cardinality `21` containing an element of order `21` is `ZMod 21`. -/
noncomputable def zmodTwentyOneEquivOfCard
    [Finite G]
    (T : G)
    (hT : addOrderOf T = 21)
    (hcard : Nat.card G = 21) :
    ZMod 21 ≃+ G := by
  letI : Fintype G := Fintype.ofFinite G
  apply AddEquiv.ofBijective (zmodGeneratorHom T hT)
  rw [Fintype.bijective_iff_injective_and_card]
  refine ⟨zmodGeneratorHom_injective T hT, ?_⟩
  simpa [Nat.card_eq_fintype_card] using hcard.symm

/--
Top-level cyclicity constructor. The explicit order-three and order-seven points
supply the lower bound `21 ≤ #G`; the preceding Block-7 theorem supplies `#G ≤ 21`.
-/
noncomputable def zmodTwentyOneEquivOfBlock7
    {A : Type*} [AddCommGroup A]
    (h21 : ∀ x : G, (21 : ℕ) • x = 0)
    [Finite (G ⧸ ThreeRange G)]
    [Finite A]
    (hmod3 : Nat.card (G ⧸ ThreeRange G) ≤ 3)
    (red : G →+ A)
    (hker7 : ∀ x : G,
      (7 : ℕ) • x = 0 → red x = 0 → x = 0)
    (hA : Nat.card A = 7)
    (T3 T7 : G)
    (h3 : addOrderOf T3 = 3)
    (h7 : addOrderOf T7 = 7) :
    ZMod 21 ≃+ G := by
  letI : Finite G :=
    finite_of_modThree_and_reduction h21 red hker7
  have hT : addOrderOf (T3 + T7) = 21 :=
    addOrderOf_add_eq_twentyOne h3 h7
  have hUpper : Nat.card G ≤ 21 :=
    card_le_twentyOne h21 hmod3 red hker7 hA
  have hLower : 21 ≤ Nat.card G := by
    exact natCard_le_of_injective
      (zmodGeneratorHom (T3 + T7) hT)
      (zmodGeneratorHom_injective (T3 + T7) hT)
  exact zmodTwentyOneEquivOfCard (T3 + T7) hT
    (le_antisymm hUpper hLower)

end CyclicTwentyOne

end N18Block7

# 3. A finite degree-two fiber table with exactly 42 slots

Using a slot type is more robust than manipulating a union of 21 finsets. Ramified fibers simply use one slot and set the other to none; empty fibers use two nones.

```javascript
namespace N18Block7

abbrev Z21 := ZMod 21
abbrev FiberSlot := Z21 × Fin 2

lemma natCard_fiberSlot : Nat.card FiberSlot = 42 := by
  simp [FiberSlot, Nat.card_zmod]

/--
A complete table for a degree-two map from `CL` to a group identified with
`ZMod 21`.

`pointAt (k,i)` is the `i`th point in the fiber above `enumE k`, if it exists.
The completeness field forces the table to include poles, ramification points,
and the two points at infinity; it is therefore a projective, not merely affine,
fiber certificate.
-/
structure DegreeTwoFiberTable
    (CQ CL E : Type*) [AddCommGroup E] where
  baseChange : CQ ↪ CL
  qPlus : CL → E
  enumE : Z21 ≃+ E
  pointAt : FiberSlot → Option CL
  pointAt_sound :
    ∀ s : FiberSlot, ∀ P : CL,
      pointAt s = some P → qPlus P = enumE s.1
  pointAt_complete :
    ∀ P : CL, ∃ i : Fin 2,
      pointAt (enumE.symm (qPlus P), i) = some P

namespace DegreeTwoFiberTable

variable {CQ CL E : Type*} [AddCommGroup E]
    (T : DegreeTwoFiberTable CQ CL E)

/-- The subtype of geometric points actually occurring in the 42 slots. -/
def GeometricCandidate : Type* :=
  {P : CL // ∃ s : FiberSlot, T.pointAt s = some P}

noncomputable def chosenSlot (P : T.GeometricCandidate) : FiberSlot :=
  Classical.choose P.property

lemma chosenSlot_spec (P : T.GeometricCandidate) :
    T.pointAt (T.chosenSlot P) = some (P : CL) :=
  Classical.choose_spec P.property

theorem chosenSlot_injective :
    Function.Injective T.chosenSlot := by
  intro P Q h
  apply Subtype.ext
  have hP := T.chosenSlot_spec P
  have hQ := T.chosenSlot_spec Q
  rw [h] at hP
  exact Option.some.inj (hP.symm.trans hQ)

noncomputable def pointToCandidate (P : CL) : T.GeometricCandidate :=
  ⟨P, by
    obtain ⟨i, hi⟩ := T.pointAt_complete P
    exact ⟨(T.enumE.symm (T.qPlus P), i), hi⟩⟩

noncomputable def pointSlot (P : CL) : FiberSlot :=
  T.chosenSlot (T.pointToCandidate P)

theorem pointSlot_injective :
    Function.Injective T.pointSlot := by
  intro P Q h
  have hc : T.pointToCandidate P = T.pointToCandidate Q :=
    T.chosenSlot_injective h
  exact congrArg Subtype.val hc

/-- All `L`-points of the curve form a finite type. -/
theorem curvePoints_finite : Finite CL :=
  Finite.of_injective T.pointSlot T.pointSlot_injective

/-- The degree-two table gives the effective bound `#C(L) ≤ 42`. -/
theorem curvePoints_card_le_42 : Nat.card CL ≤ 42 := by
  letI : Finite CL := T.curvePoints_finite
  calc
    Nat.card CL ≤ Nat.card FiberSlot :=
      natCard_le_of_injective T.pointSlot T.pointSlot_injective
    _ = 42 := natCard_fiberSlot

noncomputable def rationalPointSlot (P : CQ) : FiberSlot :=
  T.pointSlot (T.baseChange P)

theorem rationalPointSlot_injective :
    Function.Injective T.rationalPointSlot := by
  intro P Q h
  apply T.baseChange.injective
  exact T.pointSlot_injective h

/-- In the actual file `CQ` is already the subtype of rational points on `C`. -/
def C_Q_points : Set CQ := Set.univ

/-- Requested finiteness statement. -/
theorem C_Q_points_finite : T.C_Q_points.Finite := by
  letI : Finite CQ :=
    Finite.of_injective T.rationalPointSlot T.rationalPointSlot_injective
  simpa [C_Q_points] using
    (Set.finite_univ : (Set.univ : Set CQ).Finite)

/-- Effective rational-point cardinal bound. -/
theorem rationalPoints_card_le_42 : Nat.card CQ ≤ 42 := by
  letI : Finite CQ :=
    Finite.of_injective T.rationalPointSlot T.rationalPointSlot_injective
  calc
    Nat.card CQ ≤ Nat.card FiberSlot :=
      natCard_le_of_injective T.rationalPointSlot
        T.rationalPointSlot_injective
    _ = 42 := natCard_fiberSlot

end DegreeTwoFiberTable

end N18Block7
```

This argument proves finiteness by an explicit injection into a 42-element slot type. It does not infer finiteness merely from bounded exponent.

# 4. Rationality filtering and the six-cusp check

The geometric table lives over L; a second finite table records which slots descend to rational points. This makes the final theorem a finite computation.

```javascript
namespace N18Block7

structure RationalFiberCertificate
    (CQ CL E : Type*) [AddCommGroup E] where
  geom : DegreeTwoFiberTable CQ CL E

  /-- `rationalAt s = some P` means that slot `s` is the scalar extension of `P`. -/
  rationalAt : FiberSlot → Option CQ

  rationalAt_sound :
    ∀ s : FiberSlot, ∀ P : CQ,
      rationalAt s = some P →
        geom.pointAt s = some (geom.baseChange P)

  /-- Every rational point occurs in the finite table. -/
  rationalAt_complete :
    ∀ P : CQ, ∃ s : FiberSlot, rationalAt s = some P

namespace RationalFiberCertificate

variable {CQ CL E : Type*} [AddCommGroup E]
    [DecidableEq CQ]
    (R : RationalFiberCertificate CQ CL E)

/-- Deduplicated rational candidates extracted from the 42 slots. -/
noncomputable def rationalCandidateFinset : Finset CQ :=
  Finset.univ.biUnion fun s : FiberSlot =>
    (R.rationalAt s).toFinset

theorem rational_mem_candidateFinset (P : CQ) :
    P ∈ R.rationalCandidateFinset := by
  obtain ⟨s, hs⟩ := R.rationalAt_complete P
  rw [rationalCandidateFinset, Finset.mem_biUnion]
  exact ⟨s, Finset.mem_univ s, by simp [hs]⟩

/-- The candidate finset also has cardinality at most 42. -/
theorem rationalCandidateFinset_card_le_42 :
    R.rationalCandidateFinset.card ≤ 42 := by
  letI : Finite CQ :=
    Finite.of_injective R.geom.rationalPointSlot
      R.geom.rationalPointSlot_injective
  letI : Fintype CQ := Fintype.ofFinite CQ
  calc
    R.rationalCandidateFinset.card ≤ Fintype.card CQ := by
      simpa using
        Finset.card_le_card
          (Finset.subset_univ R.rationalCandidateFinset)
    _ = Nat.card CQ := by simp
    _ ≤ 42 := R.geom.rationalPoints_card_le_42

/-- A checked equality of finite tables exhausts all rational points. -/
theorem all_rational_points_mem
    (cusps : Finset CQ)
    (hcheck : R.rationalCandidateFinset = cusps) :
    ∀ P : CQ, P ∈ cusps := by
  intro P
  rw [← hcheck]
  exact R.rational_mem_candidateFinset P

end RationalFiberCertificate

end N18Block7
```
