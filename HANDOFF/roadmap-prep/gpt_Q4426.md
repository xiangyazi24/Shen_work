# 1. Complete reusable group-theory core

```javascript
import Mathlib

set_option autoImplicit false
set_option maxHeartbeats 0

namespace N18RouteCBlock5

noncomputable section

/-!
`InfinitelyNSmulDivisible n x` means that `x` has an `n^k`-th
preimage for every `k`.
-/
def InfinitelyNSmulDivisible
    {G : Type*} [AddCommGroup G] (n : ℕ) (x : G) : Prop :=
  ∀ k : ℕ, ∃ y : G, (n ^ k) • y = x

/-- The `n`-adic filtration on `G` is separated. -/
def NSeparated
    (G : Type*) [AddCommGroup G] (n : ℕ) : Prop :=
  ∀ x : G, InfinitelyNSmulDivisible n x → x = 0

/-- Witness form of weak descent.

Every element is a bounded-exponent error plus an `n`-multiple.
This is exactly the form emitted by Block 4.
-/
def WeakDescent
    (G : Type*) [AddCommGroup G] (n a : ℕ) : Prop :=
  ∀ x : G, ∃ h y : G,
    a • h = 0 ∧ x = h + n • y

/-- If `a` kills `x` and `a ∣ e`, then `e` kills `x`. -/
lemma nsmul_eq_zero_of_dvd
    {G : Type*} [AddCommGroup G]
    {a e : ℕ} (hae : a ∣ e) {x : G}
    (hx : a • x = 0) :
    e • x = 0 := by
  rcases hae with ⟨c, rfl⟩
  rw [Nat.mul_comm, mul_nsmul, hx, nsmul_zero]

/-- Iterate weak descent after multiplying by a common multiple of the
error exponent.

No finite generation is used.
-/
lemma weakDescent_iter_commonMultiple
    {G : Type*} [AddCommGroup G]
    {n a e : ℕ}
    (weak : WeakDescent G n a)
    (hae : a ∣ e) :
    ∀ k : ℕ, ∀ x : G,
      ∃ y : G, e • x = (n ^ k) • (e • y) := by
  intro k
  induction k with
  | zero =>
      intro x
      exact ⟨x, by simp⟩
  | succ k ih =>
      intro x
      obtain ⟨y, hy⟩ := ih x
      obtain ⟨h, z, hh, hz⟩ := weak y
      have heh : e • h = 0 := nsmul_eq_zero_of_dvd hae hh
      refine ⟨z, ?_⟩
      calc
        e • x = (n ^ k) • (e • y) := hy
        _ = (n ^ k) • (e • (h + n • z)) := by rw [hz]
        _ = (n ^ k) • (n • (e • z)) := by
          congr 1
          rw [nsmul_add, heh, zero_add]
          simp [mul_nsmul, Nat.mul_comm]
        _ = (n ^ (k + 1)) • (e • z) := by
          rw [pow_succ, mul_nsmul]

/-- Main common-multiple theorem.

`loc` embeds the global group into a local group. `red` is reduction.
The only local input is separatedness of `red.ker`.
-/
theorem annihilated_of_commonMultiple
    {G A R : Type*}
    [AddCommGroup G] [AddCommGroup A] [AddCommGroup R]
    (loc : G →+ A)
    (hloc : Function.Injective loc)
    (red : A →+ R)
    (n a b e : ℕ)
    (weak : WeakDescent G n a)
    (killRed : ∀ r : R, b • r = 0)
    (hae : a ∣ e)
    (hbe : b ∣ e)
    (separated : NSeparated red.ker n) :
    ∀ x : G, e • x = 0 := by
  have killRedE : ∀ r : R, e • r = 0 := by
    intro r
    exact nsmul_eq_zero_of_dvd hbe (killRed r)

  have hiter := weakDescent_iter_commonMultiple weak hae

  intro x

  let P : red.ker :=
    ⟨loc (e • x), by
      change red (loc (e • x)) = 0
      simpa only [map_nsmul] using killRedE (red (loc x))⟩

  have hdiv : InfinitelyNSmulDivisible n P := by
    intro k
    obtain ⟨y, hy⟩ := hiter k x
    let Q : red.ker :=
      ⟨loc (e • y), by
        change red (loc (e • y)) = 0
        simpa only [map_nsmul] using killRedE (red (loc y))⟩
    refine ⟨Q, ?_⟩
    apply Subtype.ext
    simpa [P, Q] using congrArg loc hy.symm

  have hP0 : P = 0 := separated P hdiv
  apply hloc
  simpa [P] using congrArg Subtype.val hP0

/-- The requested `lcm(a,b)` form. -/
theorem annihilated_of_weakDescent_and_separated
    {G A R : Type*}
    [AddCommGroup G] [AddCommGroup A] [AddCommGroup R]
    (loc : G →+ A)
    (hloc : Function.Injective loc)
    (red : A →+ R)
    (n a b : ℕ)
    (weak : WeakDescent G n a)
    (killRed : ∀ r : R, b • r = 0)
    (separated : NSeparated red.ker n) :
    ∀ x : G, Nat.lcm a b • x = 0 := by
  exact annihilated_of_commonMultiple
    loc hloc red n a b (Nat.lcm a b)
    weak killRed
    (Nat.dvd_lcm_left a b)
    (Nat.dvd_lcm_right a b)
    separated

/-- Subgroup presentation of the same theorem. -/
theorem annihilated_of_weakDescent_subgroup
    {G A R : Type*}
    [AddCommGroup G] [AddCommGroup A] [AddCommGroup R]
    (loc : G →+ A)
    (hloc : Function.Injective loc)
    (red : A →+ R)
    (H : AddSubgroup G)
    (n a b : ℕ)
    (split : ∀ x : G, ∃ h : H, ∃ y : G,
      x = (h : G) + n • y)
    (killH : ∀ h : H, a • (h : G) = 0)
    (killRed : ∀ r : R, b • r = 0)
    (separated : NSeparated red.ker n) :
    ∀ x : G, Nat.lcm a b • x = 0 := by
  apply annihilated_of_weakDescent_and_separated
    loc hloc red n a b
  · intro x
    obtain ⟨h, y, hxy⟩ := split x
    exact ⟨(h : G), y, killH h, hxy⟩
  · exact killRed
  · exact separated

end
```

# 2. Strict valuation growth implies separatedness

The formal group is used only to construct a natural-number-valued level on the nonzero formal kernel with a strict growth step.

```javascript
namespace N18RouteCBlock5

noncomputable section

/-- An abstract strictly increasing `n`-adic filtration.

The step theorem deliberately assumes `n • x ≠ 0`. This is the clean
finite-valued treatment of the order-3 torsion point in the formal kernel.
-/
structure StrictNSmulFiltration
    (K : Type*) [AddCommGroup K] (n : ℕ) where
  level : K → ℕ
  step : ∀ {x : K},
    x ≠ 0 → n • x ≠ 0 →
    level x + 1 ≤ level (n • x)

lemma StrictNSmulFiltration.iterate
    {K : Type*} [AddCommGroup K]
    {n : ℕ}
    (D : StrictNSmulFiltration K n)
    {x : K} :
    ∀ k : ℕ,
      (n ^ k) • x ≠ 0 →
      D.level x + k ≤ D.level ((n ^ k) • x) := by
  intro k
  induction k with
  | zero =>
      intro _
      simp
  | succ k ih =>
      intro hfinal
      have hprev : (n ^ k) • x ≠ 0 := by
        intro hz
        apply hfinal
        simpa [pow_succ, mul_nsmul, Nat.mul_comm, hz]
      have hstepNe : n • ((n ^ k) • x) ≠ 0 := by
        simpa [pow_succ, mul_nsmul, Nat.mul_comm] using hfinal
      have hi := ih hprev
      have hs := D.step hprev hstepNe
      calc
        D.level x + (k + 1) = (D.level x + k) + 1 := by omega
        _ ≤ D.level ((n ^ k) • x) + 1 := Nat.add_le_add_right hi 1
        _ ≤ D.level (n • ((n ^ k) • x)) := hs
        _ = D.level ((n ^ (k + 1)) • x) := by
          congr 1
          simp [pow_succ, mul_nsmul, Nat.mul_comm]

/-- Strict level growth rules out every nonzero infinitely `n`-divisible
point. This theorem includes no topology and no finite generation. -/
theorem StrictNSmulFiltration.separated
    {K : Type*} [AddCommGroup K]
    {n : ℕ}
    (D : StrictNSmulFiltration K n) :
    NSeparated K n := by
  intro P hdiv
  by_contra hP
  let N : ℕ := D.level P + 1
  obtain ⟨Q, hQ⟩ := hdiv N
  have hfinal : (n ^ N) • Q ≠ 0 := by
    rw [hQ]
    exact hP
  have hi := D.iterate N hfinal
  dsimp [N] at hi
  omega

end
```

# 5. Fully proved transfer and push-pull lemmas

```javascript
namespace N18RouteCBlock5

/-- An annihilator transfers backward through an injective homomorphism. -/
theorem annihilator_transfer_injective
    {A B : Type*} [AddCommGroup A] [AddCommGroup B]
    (f : A →+ B)
    (hf : Function.Injective f)
    (m : ℕ)
    (killB : ∀ b : B, m • b = 0) :
    ∀ a : A, m • a = 0 := by
  intro a
  apply hf
  simpa only [map_nsmul, map_zero] using killB (f a)

/-- Conservative transfer through a dual-isogeny identity
`phi(phihat Q)=n•Q`. -/
theorem annihilator_transfer_via_dual_comp
    {A B : Type*} [AddCommGroup A] [AddCommGroup B]
    (phi : A →+ B)
    (phihat : B →+ A)
    (n m : ℕ)
    (hcomp : ∀ Q : B, phi (phihat Q) = n • Q)
    (killA : ∀ P : A, m • P = 0) :
    ∀ Q : B, (n * m) • Q = 0 := by
  intro Q
  calc
    (n * m) • Q = m • (n • Q) := by
      rw [Nat.mul_comm, mul_nsmul]
    _ = m • phi (phihat Q) := by rw [← hcomp Q]
    _ = phi (m • phihat Q) := by rw [map_nsmul]
    _ = 0 := by rw [killA, map_zero]

/-- If both factors are killed by `m` and `Psi(Phi D)=2D`, then the
source is killed by `2m`. -/
theorem pushPull_annihilator
    {J A B : Type*}
    [AddCommGroup J] [AddCommGroup A] [AddCommGroup B]
    (Phi : J →+ A × B)
    (Psi : A × B →+ J)
    (m : ℕ)
    (killA : ∀ a : A, m • a = 0)
    (killB : ∀ b : B, m • b = 0)
    (pushPull : ∀ D : J, Psi (Phi D) = 2 • D) :
    ∀ D : J, (2 * m) • D = 0 := by
  intro D
  have hpair : m • Phi D = 0 := by
    apply Prod.ext
    · simpa using killA (Phi D).1
    · simpa using killB (Phi D).2
  calc
    (2 * m) • D = m • (2 • D) := by
      rw [Nat.mul_comm, mul_nsmul]
    _ = m • Psi (Phi D) := by rw [← pushPull D]
    _ = Psi (m • Phi D) := by rw [map_nsmul]
    _ = 0 := by rw [hpair, map_zero]

end N18RouteCBlock5
```
