/-- If `Q=(x,y)` and `x≠0`, these are the coordinates of `Q+(0,0)`. -/
def tateAddMarkedXY (b : ℚ) (Q : ℚ × ℚ) : ℚ × ℚ :=
  (b * Q.2 / Q.1 ^ 2,
   b ^ 2 * (Q.1 ^ 2 - Q.2) / Q.1 ^ 3)

/-- Coordinates of `(k+3)P`, beginning with `3P=(c,b-c)`. -/
def tateFromThree (b c : ℚ) (k : ℕ) : ℚ × ℚ :=
  ((tateAddMarkedXY b)^[k]) (c, b - c)

/-! ========================================================================
    3. Raw affine model of X_1(18)
    ======================================================================== -/

/-- Raw order-18 equation in `r=b/c`, `s=c²/(b-c)`. -/
def F18 (r s : ℚ) : ℚ :=
    r ^ 4 * s ^ 3 - 6 * r ^ 4 * s ^ 2 + 9 * r ^ 4 * s - r ^ 4
  + r ^ 3 * s ^ 5 - 7 * r ^ 3 * s ^ 4 + 20 * r ^ 3 * s ^ 3
  - 19 * r ^ 3 * s ^ 2 - 8 * r ^ 3 * s + r ^ 3
  + r ^ 2 * s ^ 4 - 11 * r ^ 2 * s ^ 3 + 28 * r ^ 2 * s ^ 2
  + r * s ^ 4 - 5 * r * s ^ 3 - 8 * r * s ^ 2
  + s ^ 4 + s ^ 3 + s ^ 2

/-- Exact denominator-cleared pullback identity. This is a pure `ring`
certificate, not a moduli theorem. -/
theorem T18_pullback_F18
    (b c : ℚ) (hc : c ≠ 0) (hbc : b - c ≠ 0) :
    T18 b c =
      c ^ 4 * (b - c) ^ 5 * F18 (b / c) (c ^ 2 / (b - c)) := by
  field_simp [hc, hbc]
  ring

  r : ℚ
  s : ℚ

  r_ne_zero : r ≠ 0
  r_ne_one : r ≠ 1
  s_ne_one : s ≠ 1

/-- Tate parameters to the raw affine modular curve. -/
noncomputable def rawOfTate (p : Tate18Params) : Raw18Point := by
  refine
    { r := p.b / p.c
      s := p.c ^ 2 / (p.b - p.c)
      onCurve := ?_
      r_ne_zero := div_ne_zero p.hb p.hc
      r_ne_one := ?_
      s_ne_one := ?_ }
  · have hpull := T18_pullback_F18 p.b p.c p.hc p.hbc
    have hz :
        p.c ^ 4 * (p.b - p.c) ^ 5 *
          F18 (p.b / p.c) (p.c ^ 2 / (p.b - p.c)) = 0 := by
      rw [← hpull, p.hprimitive]
    have hfac : p.c ^ 4 * (p.b - p.c) ^ 5 ≠ 0 :=
      mul_ne_zero (pow_ne_zero _ p.hc) (pow_ne_zero _ p.hbc)
    exact (mul_eq_zero.mp hz).resolve_left hfac
  · intro h
    apply p.hbc
    field_simp [p.hc] at h
    linarith
  · intro h
    apply p.hlower
    field_simp [p.hbc] at h
    nlinarith

def Wden (r : ℚ) : ℚ := r ^ 2 * (r - 1) ^ 7

def rawW (R : Raw18Point) : ℚ := Wnum R.r R.s / Wden R.r

def rawU (R : Raw18Point) : ℚ := 1 + rawW R

/-- Optimized quadratic model. -/

  v ^ 2 + optA u * v + 2 * u = 0

  (s - r) * w ^ 2 + r * (s - 1) * w + (r - 1)

  N ^ 2 + (w ^ 4 + 2 * w ^ 3 + 3 * w ^ 2 + 5 * w + 3) * N * D

/-! ========================================================================
    5. Optimized model <-> the sextic C
    ======================================================================== -/

/-- Affine sextic model of `X_1(18)`. -/
def f18 (x : ℚ) : ℚ :=
  x ^ 6 + 4 * x ^ 5 + 10 * x ^ 4 + 10 * x ^ 3 +
    5 * x ^ 2 + 2 * x + 1

/-- Rational points on the smooth projective monic degree-six model.
The two constructors at infinity are the two rational points above infinity. -/
inductive CPoint where
  | infinityPlus
  | infinityMinus
  | affine (x y : ℚ) (onCurve : y ^ 2 = f18 x)

/-- The six rational cusps in this model:

def IsCusp : CPoint → Prop

  | .infinityMinus => True
  | .affine x y _ =>
      (x = 0 ∧ (y = 1 ∨ y = -1)) ∨
      (x = -1 ∧ (y = 1 ∨ y = -1))

  calc
    (2 * v + optA u) ^ 2 - f18 (-u)

  · exact hu1 (by linarith [h1.1])

  rw [hv, hxy] at hc

    ¬ IsCusp (tateToCurve p) := by
  apply curveOfOptimized_not_cusp
  · exact rawU_ne_zero (rawOfTate p)
  · exact rawU_ne_one (rawOfTate p)

# What is genuinely deep and what is finite

The draft has four marked seams, but only the first is conceptually deep.

The following portions are already fully elementary in the draft:

- the Tate discriminant identity;

- the exact T18/F18 pullback;

- all open-condition deductions needed for r≠0, r≠1, s≠1;

- the proof that the optimized coordinate u is neither 0 nor 1;

- the optimized-model/sextic isomorphism;

- the noncusp proof;

- the final composition with Block 7.

# Integration cautions

1. Do not replace exact order by [18]P=O. The fields nine_ne_zero, six_ne_zero, and hlower are load-bearing. Without them, lower-order components and cusps enter the parameter curve.
