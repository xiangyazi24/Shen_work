/-! ## 6. Fully filled weak-descent chase -/

/-- The actual Block-4 output:

    E0(L) = <T> + 3 E0(L).

It is expressed without constructing a quotient group. -/
theorem weakThreeDescent
    (I : IsogenyData (L := L) a)
    (D : KummerData (L := L) a I)
    (P : EPoint (L := L)) :
    ∃ n : Fin 3, ∃ Q : EPoint (L := L),
      P = n.val • T (L := L) + 3 • Q := by
  have hdual : passDual (D.deltaHat P) := D.deltaHat_local P
  obtain ⟨k, hk⟩ := (passDual_iff_all (D.deltaHat P)).mp hdual

  obtain ⟨n, hn⟩ := dualLine_represented k

  have hTclass :
      D.deltaHat (n.val • T (L := L)) = mkDual 0 0 k 0 := by
    simpa only [map_nsmul, D.deltaHat_T] using hn

  have hkerHat :
      D.deltaHat (P - n.val • T (L := L)) = 0 := by
    rw [map_sub, hk, hTclass, sub_self]

  obtain ⟨R, hR⟩ :=
    (D.kernelHat (P - n.val • T (L := L))).mp hkerHat

  have hphi : passPhi (D.deltaPhi R) := D.deltaPhi_local R
  have hkerPhi : D.deltaPhi R = 0 :=
    (passPhi_iff_all (D.deltaPhi R)).mp hphi

  obtain ⟨Q, hQ⟩ := (D.kernelPhi R).mp hkerPhi

  refine ⟨n, Q, ?_⟩
  calc
    P = n.val • T (L := L) + I.phihat R := by
      rw [hR]
      abel
    _ = n.val • T (L := L) + I.phihat (I.phi Q) := by
      rw [hQ]
    _ = n.val • T (L := L) + 3 • Q := by
      rw [I.phihat_phi]

    (D : KummerData (L := L) a I)
    (P : EPoint (L := L)) :
    ∃ h : EPoint (L := L),
      3 • h = 0 ∧
      ∃ Q : EPoint (L := L), P = h + 3 • Q := by
  obtain ⟨n, Q, hPQ⟩ := weakThreeDescent I D P
  refine ⟨n.val • T (L := L), ?_, Q, hPQ⟩
  simpa [mul_nsmul, Nat.mul_comm, I.T_order]
