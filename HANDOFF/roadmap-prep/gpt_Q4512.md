```javascript
import Mathlib.RingTheory.Ideal.Operations
import Mathlib.Tactic

set_option autoImplicit false

namespace FLT.CyclicExclusion.N18.MumfordIdeal

open Set

variable {A : Type*} [CommRing A]

def mumfordIdeal (U Y V : A) : Ideal A :=
  Ideal.span {U, Y - V}

def mumfordConjIdeal (U Y V : A) : Ideal A :=
  Ideal.span {U, Y + V}

/-- Adding a multiple of the first generator to the second generator
    does not change a two-generated ideal. -/
theorem span_pair_add_mul_first (U T Q : A) :
    Ideal.span {U, T + Q * U} = Ideal.span {U, T} := by
  ext Z
  simp only [Ideal.mem_span_pair]
  constructor
  · rintro ⟨a, b, rfl⟩
    refine ⟨a + b * Q, b, ?_⟩
    ring
  · rintro ⟨a, b, rfl⟩
    refine ⟨a - b * Q, b, ?_⟩
    ring

/-- The concrete inverse-ideal calculation used for hyperelliptic
    negation.  The Bézout hypothesis is the primitive Mumford condition. -/
theorem mumfordIdeal_mul_conj
    (U V Y W : A)
    (hrel : Y ^ 2 - V ^ 2 = U * W)
    (hprimitive : ∃ a b c : A,
      a * U + b * (2 * V) + c * W = 1) :
    mumfordIdeal U Y V * mumfordConjIdeal U Y V =
      Ideal.span {U} := by
  unfold mumfordIdeal mumfordConjIdeal
  apply le_antisymm

  · rw [Ideal.span_pair_mul_span_pair]
    refine Ideal.span_le.2 ?_
    intro Z hZ
    simp only [Set.mem_insert_iff, Set.mem_singleton_iff] at hZ
    rcases hZ with rfl | rfl | rfl | rfl
    · exact Ideal.mem_span_singleton.mpr ⟨U, rfl⟩
    · exact Ideal.mem_span_singleton.mpr ⟨Y + V, rfl⟩
    · exact Ideal.mem_span_singleton.mpr ⟨Y - V, by ring⟩
    · refine Ideal.mem_span_singleton.mpr ⟨W, ?_⟩
      calc
        (Y - V) * (Y + V) = Y ^ 2 - V ^ 2 := by ring
        _ = U * W := hrel

  · rw [Ideal.span_singleton_le_iff_mem]
    obtain ⟨a, b, c, habc⟩ := hprimitive

    let I : Ideal A := Ideal.span {U, Y - V}
    let J : Ideal A := Ideal.span {U, Y + V}

    have hU_I : U ∈ I := by
      exact Ideal.subset_span (by simp [I])
    have hU_J : U ∈ J := by
      exact Ideal.subset_span (by simp [J])
    have hminus_I : Y - V ∈ I := by
      exact Ideal.subset_span (by simp [I])
    have hplus_J : Y + V ∈ J := by
      exact Ideal.subset_span (by simp [J])

    have hUU : U * U ∈ I * J :=
      Ideal.mul_mem_mul hU_I hU_J
    have hUplus : U * (Y + V) ∈ I * J :=
      Ideal.mul_mem_mul hU_I hplus_J
    have hminusU : (Y - V) * U ∈ I * J :=
      Ideal.mul_mem_mul hminus_I hU_J
    have hminusPlus : (Y - V) * (Y + V) ∈ I * J :=
      Ideal.mul_mem_mul hminus_I hplus_J

    have hdiff : U * (Y + V) - (Y - V) * U ∈ I * J :=
      (I * J).sub_mem hUplus hminusU

    have hsum :
        a * (U * U)
          + b * (U * (Y + V) - (Y - V) * U)
          + c * ((Y - V) * (Y + V)) ∈ I * J := by
      exact (I * J).add_mem
        ((I * J).add_mem
          ((I * J).mul_mem_left a hUU)
          ((I * J).mul_mem_left b hdiff))
        ((I * J).mul_mem_left c hminusPlus)

    have heq :
        a * (U * U)
          + b * (U * (Y + V) - (Y - V) * U)
          + c * ((Y - V) * (Y + V)) = U := by
      calc
        a * (U * U)
            + b * (U * (Y + V) - (Y - V) * U)
            + c * ((Y - V) * (Y + V))
            = a * U ^ 2 + b * (2 * U * V)
                + c * (Y ^ 2 - V ^ 2) := by ring
        _ = a * U ^ 2 + b * (2 * U * V) + c * (U * W) := by
          rw [hrel]
        _ = U * (a * U + b * (2 * V) + c * W) := by ring
        _ = U := by rw [habc, mul_one]

    change U ∈ I * J
    rw [← heq]
    exact hsum
```

## 3.1 Divisors as a free abelian group

```javascript
import Mathlib.GroupTheory.FreeAbelianGroup

namespace FLT.CyclicExclusion.N18.DeckTrace

abbrev Divisor (X : Type*) := FreeAbelianGroup X

noncomputable def pushDiv {X Y : Type*} (q : X → Y) :
    Divisor X →+ Divisor Y :=
  FreeAbelianGroup.lift (fun P => FreeAbelianGroup.of (q P))

noncomputable def pullDiv {X Y : Type*}
    (fiber : Y → Divisor X) :
    Divisor Y →+ Divisor X :=
  FreeAbelianGroup.lift fiber

noncomputable def deckDiv {X : Type*} (rho : X → X) :
    Divisor X →+ Divisor X :=
  FreeAbelianGroup.lift (fun P => FreeAbelianGroup.of (rho P))

/-- Raw divisor trace identity, proved on the free generators. -/
theorem pull_push_divisor
    {X Y : Type*}
    (q : X → Y) (rho : X → X)
    (fiber : Y → Divisor X)
    (hfiber : ∀ P : X,
      fiber (q P) = FreeAbelianGroup.of P
        + FreeAbelianGroup.of (rho P)) :
    (pullDiv fiber).comp (pushDiv q) =
      AddMonoidHom.id (Divisor X) + deckDiv rho := by
  apply FreeAbelianGroup.lift_ext
  intro P
  simp [pullDiv, pushDiv, deckDiv, hfiber]
```

```javascript
noncomputable def orbitFiber

  Quotient.lift
    (fun P => FreeAbelianGroup.of P + FreeAbelianGroup.of (rho P))

      intro P Q hPQ
      rcases hPQ with rfl | hPQ
      · rfl
      · subst Q
```
