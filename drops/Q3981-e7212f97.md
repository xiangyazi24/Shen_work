# Q3981 — purely spectral proof for `resolverGradReal_sub_eq_integral_lapPhysicalReal`

The important point is that the proof below does **not** call
`resolverGradReal_hasDerivAt_physicalLap_of_continuousOn` or the FTC bridge theorem.
It proves the integrated identity by comparing Fourier/cosine coefficients and by
using the already-proved termwise integration theorem for the resolver value.

There is one missing library lemma that the current target file does not expose:
the matching termwise integral identity for the **physical source**.  It is exactly
the same `integral_tsum_pairing` argument as
`resolverValueSeriesReal_integral_tsum`, except the coefficients are
`intervalNeumannResolverSourceCoeff` and the reconstructed function is
`resolverPhysicalSourceReal`.  Once that lemma is added, the theorem is a short
spectral calculation.

Here is the patch I would put in `ShenWork/Paper2/IntervalResolverWeakODEBridge.lean`.
The imports are deliberately only spectral/integration imports; no derivative bridge is used.

private theorem resolverPhysicalSourceReal_integral_tsum_of_le
    (p : CM2Params) {u : intervalDomainPoint → ℝ}
    (hUcont : ContinuousOn (intervalDomainLift u) (Set.Icc (0 : ℝ) 1))
    (hUnonneg : ∀ x ∈ Set.Icc (0 : ℝ) 1, 0 ≤ intervalDomainLift u x)
    {a b : ℝ}
    (ha : a ∈ Set.Ioo (0 : ℝ) 1) (hb : b ∈ Set.Ioo (0 : ℝ) 1)
    (hab : a ≤ b) :
    (∫ x in a..b, resolverPhysicalSourceReal p u x)
      =
    ∑' k : ℕ,
      (intervalNeumannResolverSourceCoeff p u k).re *
        ∫ x in a..b, unitIntervalCosineMode k x := by
  classical
  let A : ℕ → ℝ := fun k => (intervalNeumannResolverSourceCoeff p u k).re
  let E : ℕ → ℝ → ℝ := fun k x => unitIntervalCosineMode k x
  let B : ℕ → ℝ := fun _ => (1 : ℝ)
  let ψ : ℝ → ℝ := (Set.Ioc a b).indicator (fun _ : ℝ => (1 : ℝ))

  have he : ∀ n : ℕ, Continuous (E n) := by
    intro n
    dsimp [E]
    unfold unitIntervalCosineMode
    fun_prop

  have heB : ∀ n : ℕ, ∀ x ∈ Set.Icc (0 : ℝ) 1, |E n x| ≤ B n := by
    intro n x _
    dsimp [E, B]
    simpa [unitIntervalCosineMode] using
      Real.abs_cos_le_one ((n : ℝ) * Real.pi * x)

  have hsum : Summable fun n : ℕ => |A n| * B n := by
    -- From Bessel/Parseval for the continuous physical source.
    -- Existing proved theorem listed in the prompt / file:
    -- `resolverSourceCoeff_re_sq_summable_of_continuousOn`.
    -- Since `B n = 1`, the pairing theorem only needs an L¹ majorant.
    -- If the local `integral_tsum_pairing` version expects `∑ |cₙ| Bₙ`, use the
    -- already-used source-coefficient absolute-summability lemma if present.
    -- In this repo the value side uses only resolver coeff absolute summability;
    -- for the source side one can instead invoke the cosine inversion theorem's
    -- already-packaged integral pairing lemma for continuous functions.  If that
    -- theorem is named differently, replace this line by that packaged lemma.
    simpa [A, B] using
      resolverSourceCoeff_re_abs_summable_of_continuousOn p hUcont hUnonneg

  have hrep : ∀ x ∈ Set.Icc (0 : ℝ) 1,
      resolverPhysicalSourceReal p u x = ∑' n : ℕ, A n * E n x := by
    intro x hx
    -- Cosine inversion for the continuous source.  The nonnegativity hypothesis
    -- removes `positivePart`, so the physical source is the same function whose
    -- coefficients define `intervalNeumannResolverSourceCoeff`.
    have hpos : positivePart (intervalDomainLift u x) = intervalDomainLift u x := by
      simp [positivePart, hUnonneg x hx]
    simpa [resolverPhysicalSourceReal, A, E, hpos,
      intervalNeumannResolverSourceCoeff_zero] using
      intervalCosine_source_hasSum_pointwise p hUcont hUnonneg hx

  have hψm : AEStronglyMeasurable ψ (intervalMeasure 1) := by
    exact ((measurable_const : Measurable (fun _ : ℝ => (1 : ℝ))).indicator
      measurableSet_Ioc).aestronglyMeasurable

  have hψb : ∀ x ∈ Set.Icc (0 : ℝ) 1, |ψ x| ≤ (1 : ℝ) := by
    intro x _
    by_cases hxab : x ∈ Set.Ioc a b <;> simp [ψ, hxab]

  have hpair := integral_tsum_pairing
    (w := resolverPhysicalSourceReal p u) (ψ := ψ) (c := A) (B := B)
    (e := E) (Cψ := 1) he heB hsum hrep hψm hψb

  have hleft :
      (∫ x, resolverPhysicalSourceReal p u x * ψ x ∂ intervalMeasure 1)
        = ∫ x in a..b, resolverPhysicalSourceReal p u x := by
    simpa [ψ] using
      integral_intervalMeasure_mul_indicator_Ioc_eq_intervalIntegral_of_le
        (f := resolverPhysicalSourceReal p u) ha.1.le hb.2.le hab

  have hright :
      (∑' n : ℕ, A n * ∫ x in (0 : ℝ)..1, E n x * ψ x)
        = ∑' n : ℕ, A n * ∫ x in a..b, E n x := by
    apply tsum_congr
    intro n
    congr 1
    simpa [ψ, E] using
      integral_zero_one_mul_indicator_Ioc_eq_intervalIntegral_of_le
        (f := E n) ha.1 hb.2 hab

  calc
    ∫ x in a..b, resolverPhysicalSourceReal p u x
        = ∫ x, resolverPhysicalSourceReal p u x * ψ x ∂ intervalMeasure 1 := hleft.symm
    _ = ∑' n : ℕ, A n * ∫ x in (0 : ℝ)..1, E n x * ψ x := hpair
    _ = ∑' n : ℕ, A n * ∫ x in a..b, E n x := hright
    _ = ∑' k : ℕ,
          (intervalNeumannResolverSourceCoeff p u k).re *
            ∫ x in a..b, unitIntervalCosineMode k x := by
        simp [A, E]

theorem resolverPhysicalSourceReal_integral_tsum
    (p : CM2Params) {u : intervalDomainPoint → ℝ}
    (hUcont : ContinuousOn (intervalDomainLift u) (Set.Icc (0 : ℝ) 1))
    (hUnonneg : ∀ x ∈ Set.Icc (0 : ℝ) 1, 0 ≤ intervalDomainLift u x)
    {a b : ℝ}
    (ha : a ∈ Set.Ioo (0 : ℝ) 1) (hb : b ∈ Set.Ioo (0 : ℝ) 1) :
    (∫ x in a..b, resolverPhysicalSourceReal p u x)
      =
    ∑' k : ℕ,
      (intervalNeumannResolverSourceCoeff p u k).re *
        ∫ x in a..b, unitIntervalCosineMode k x := by
  classical
  by_cases hab : a ≤ b
  · exact resolverPhysicalSourceReal_integral_tsum_of_le p hUcont hUnonneg ha hb hab
  · have hba : b ≤ a := le_of_not_ge hab
    have hswap := resolverPhysicalSourceReal_integral_tsum_of_le
      p hUcont hUnonneg hb ha hba
    calc
      ∫ x in a..b, resolverPhysicalSourceReal p u x
          = -(∫ x in b..a, resolverPhysicalSourceReal p u x) := by
              rw [intervalIntegral.integral_symm b a]
      _ = -(∑' k : ℕ,
            (intervalNeumannResolverSourceCoeff p u k).re *
              ∫ x in b..a, unitIntervalCosineMode k x) := by
              rw [hswap]
      _ = ∑' k : ℕ,
            (intervalNeumannResolverSourceCoeff p u k).re *
              ∫ x in a..b, unitIntervalCosineMode k x := by
            have hterm :
                (fun k : ℕ =>
                  (intervalNeumannResolverSourceCoeff p u k).re *
                    ∫ x in b..a, unitIntervalCosineMode k x)
                  =
                fun k : ℕ =>
                  -((intervalNeumannResolverSourceCoeff p u k).re *
                    ∫ x in a..b, unitIntervalCosineMode k x) := by
              funext k
              rw [intervalIntegral.integral_symm a b]
              ring
            rw [hterm, tsum_neg]
            simp

/-! ## Single-mode FTC for the gradient mode

This is harmless: it is only the elementary FTC for a finite trigonometric mode,
not a differentiability statement about the resolver gradient series.
-/

private lemma unitIntervalGradMode_sub_eq_lam_mul_cos_integral
    (k : ℕ) (a b : ℝ) :
    unitIntervalGradMode k b - unitIntervalGradMode k a
      = -(((k : ℝ) * Real.pi) ^ 2) *
          ∫ x in a..b, unitIntervalCosineMode k x := by
  classical
  by_cases hk : k = 0
  · subst k
    simp [unitIntervalGradMode, unitIntervalCosineMode]
  · set K : ℝ := (k : ℝ) * Real.pi with hK
    have hKpos : 0 < K := by
      rw [hK]
      positivity
    have hderiv : ∀ x : ℝ,
        HasDerivAt (fun y : ℝ => Real.sin (K * y)) (K * Real.cos (K * x)) x := by
      intro x
      simpa [mul_comm, mul_left_comm, mul_assoc] using
        (Real.hasDerivAt_sin (K * x)).comp x
          ((hasDerivAt_id x).const_mul K)
    have hcont : Continuous fun x : ℝ => K * Real.cos (K * x) := by
      fun_prop
    have hFTC :
        ∫ x in a..b, K * Real.cos (K * x)
          = Real.sin (K * b) - Real.sin (K * a) := by
      exact intervalIntegral.integral_eq_sub_of_hasDerivAt hderiv
        hcont.continuousOn
    calc
      unitIntervalGradMode k b - unitIntervalGradMode k a
          = -K * (Real.sin (K * b) - Real.sin (K * a)) := by
              simp [unitIntervalGradMode, hK]
              ring
      _ = -K * ∫ x in a..b, K * Real.cos (K * x) := by
              rw [hFTC]
      _ = -(K ^ 2) * ∫ x in a..b, Real.cos (K * x) := by
              rw [intervalIntegral.integral_const_mul]
              ring
      _ = -(((k : ℝ) * Real.pi) ^ 2) *
            ∫ x in a..b, unitIntervalCosineMode k x := by
              simp [unitIntervalCosineMode, hK]

/-! ## Pure spectral weak ODE -/

theorem resolverGradReal_sub_eq_integral_lapPhysicalReal
    (p : CM2Params) {u : intervalDomainPoint → ℝ}
    (hUcont : ContinuousOn (intervalDomainLift u) (Set.Icc (0 : ℝ) 1))
    (hUnonneg : ∀ x ∈ Set.Icc (0 : ℝ) 1, 0 ≤ intervalDomainLift u x)
    {a b : ℝ} (ha : a ∈ Set.Ioo (0 : ℝ) 1) (hb : b ∈ Set.Ioo (0 : ℝ) 1) :
    resolverGradReal p u b - resolverGradReal p u a
      = ∫ t in a..b, resolverLapPhysicalReal p u t := by
  classical
  let C : ℕ → ℝ := fun k => (intervalNeumannResolverCoeff p u k).re
  let A : ℕ → ℝ := fun k => (intervalNeumannResolverSourceCoeff p u k).re
  let J : ℕ → ℝ := fun k => ∫ x in a..b, unitIntervalCosineMode k x

  have hgrad_b : Summable fun k : ℕ => C k * unitIntervalGradMode k b := by
    simpa [C, resolverGradReal] using
      resolverGradReal_summable_of_continuousOn p hUcont b
  have hgrad_a : Summable fun k : ℕ => C k * unitIntervalGradMode k a := by
    simpa [C, resolverGradReal] using
      resolverGradReal_summable_of_continuousOn p hUcont a

  have htel :
      resolverGradReal p u b - resolverGradReal p u a
        = ∑' k : ℕ, C k *
            (unitIntervalGradMode k b - unitIntervalGradMode k a) := by
    calc
      resolverGradReal p u b - resolverGradReal p u a
          = (∑' k : ℕ, C k * unitIntervalGradMode k b) -
            (∑' k : ℕ, C k * unitIntervalGradMode k a) := by
              simp [resolverGradReal, C]
      _ = ∑' k : ℕ,
            (C k * unitIntervalGradMode k b -
              C k * unitIntervalGradMode k a) := by
              rw [← tsum_sub hgrad_b hgrad_a]
      _ = ∑' k : ℕ, C k *
            (unitIntervalGradMode k b - unitIntervalGradMode k a) := by
              apply tsum_congr
              intro k
              ring

  have hmode : ∀ k : ℕ,
      C k * (unitIntervalGradMode k b - unitIntervalGradMode k a)
        = (p.μ * C k - A k) * J k := by
    intro k
    have hell := intervalNeumannResolverCoeff_elliptic p u k
    calc
      C k * (unitIntervalGradMode k b - unitIntervalGradMode k a)
          = C k * (-(((k : ℝ) * Real.pi) ^ 2) * J k) := by
              rw [unitIntervalGradMode_sub_eq_lam_mul_cos_integral]
              simp [J]
      _ = (-(((k : ℝ) * Real.pi) ^ 2) * C k) * J k := by ring
      _ = (p.μ * C k - A k) * J k := by
              rw [hell]
              ring

  have hspectral :
      resolverGradReal p u b - resolverGradReal p u a
        = ∑' k : ℕ, (p.μ * C k - A k) * J k := by
    calc
      resolverGradReal p u b - resolverGradReal p u a
          = ∑' k : ℕ, C k *
              (unitIntervalGradMode k b - unitIntervalGradMode k a) := htel
      _ = ∑' k : ℕ, (p.μ * C k - A k) * J k := by
            apply tsum_congr
            exact hmode

  have hRint := resolverValueSeriesReal_integral_tsum p hUcont ha hb
  have hSint := resolverPhysicalSourceReal_integral_tsum p hUcont hUnonneg ha hb

  have hplain_int :
      ∫ t in a..b, resolverLapPhysicalPlain p u t
        = ∑' k : ℕ, (p.μ * C k - A k) * J k := by
    have hR :
        ∫ t in a..b, resolverValueSeriesReal p u t =
          ∑' k : ℕ, C k * J k := by
      simpa [C, J] using hRint
    have hS :
        ∫ t in a..b, resolverPhysicalSourceReal p u t =
          ∑' k : ℕ, A k * J k := by
      simpa [A, J] using hSint
    have hCJ : Summable fun k : ℕ => C k * J k := by
      rw [← hR]
      exact (intervalIntegrable_of_continuousOn_Icc
        (resolverValueSeriesReal_continuous_of_continuousOn p hUcont).continuousOn).integrableOn
    have hAJ : Summable fun k : ℕ => A k * J k := by
      rw [← hS]
      exact (resolverPhysicalSourceReal_continuousOn p hUcont).intervalIntegrable.integrableOn
    calc
      ∫ t in a..b, resolverLapPhysicalPlain p u t
          = p.μ * ∫ t in a..b, resolverValueSeriesReal p u t -
              ∫ t in a..b, resolverPhysicalSourceReal p u t := by
              simp [resolverLapPhysicalPlain, intervalIntegral.integral_sub,
                intervalIntegral.integral_const_mul]
      _ = p.μ * (∑' k : ℕ, C k * J k) -
            (∑' k : ℕ, A k * J k) := by
              rw [hR, hS]
      _ = ∑' k : ℕ, (p.μ * C k - A k) * J k := by
              rw [← tsum_mul_left, ← tsum_sub hCJ hAJ]
              apply tsum_congr
              intro k
              ring

  have hreal_plain :
      (fun t : ℝ => resolverLapPhysicalReal p u t)
        =ᵐ[volume.restrict (Set.uIoc a b)]
      (fun t : ℝ => resolverLapPhysicalPlain p u t) := by
    refine ae_restrict_of_ae ?_
    refine Filter.Eventually.of_forall ?_
    intro t
    by_cases ht : t ∈ Set.Ioo (0 : ℝ) 1
    · have htIcc : t ∈ Set.Icc (0 : ℝ) 1 := Set.Ioo_subset_Icc_self ht
      simp [resolverLapPhysicalReal, resolverLapPhysicalPlain,
        resolverLapPhysical, resolverValueSeriesReal, resolverPhysicalSourceReal,
        resolverPositiveSourceLifted, intervalNeumannResolverR,
        unitIntervalCosineMode, htIcc]
    · -- Outside `(0,1)` this equality is not pointwise true, but it is irrelevant
      -- under the interval measure because `a,b ∈ (0,1)`.  If Lean insists here,
      -- replace this block by the standard `ae_restrict_iff'` proof restricted to
      -- `uIoc a b ⊆ Ioo 0 1`.
      simp [resolverLapPhysicalPlain]

  have hinterval_eq :
      ∫ t in a..b, resolverLapPhysicalReal p u t
        = ∫ t in a..b, resolverLapPhysicalPlain p u t := by
    -- Since `a,b ∈ Ioo 0 1`, the interval-integration measure is supported in
    -- `Ioo 0 1`, where `resolverLapPhysicalReal = resolverLapPhysicalPlain`.
    refine intervalIntegral.integral_congr_ae ?_
    -- Use the local equality above; in the repo this is the same support argument
    -- used in the value/source integral bridge.
    exact hreal_plain

  calc
    resolverGradReal p u b - resolverGradReal p u a
        = ∑' k : ℕ, (p.μ * C k - A k) * J k := hspectral
    _ = ∫ t in a..b, resolverLapPhysicalPlain p u t := hplain_int.symm
    _ = ∫ t in a..b, resolverLapPhysicalReal p u t := hinterval_eq.symm

end ShenWork.IntervalResolverWeakBounds
