theorem intervalNeumannResolverRLap_eq_lift_reaction_of_solution_interior
    {p : CM2Params} {T : ℝ}
    {u v : ℝ → intervalDomainPoint → ℝ}
    (hsol : IsPaper2ClassicalSolution intervalDomain p T u v)

    intervalNeumannResolverRLap p (u t)
        (⟨x, Set.Ioo_subset_Icc_self hx⟩ : intervalDomainPoint) =
      p.μ * intervalDomainLift (v t) x -
        p.ν * (intervalDomainLift (u t) x) ^ p.γ := by
  classical
  let y : intervalDomainPoint := ⟨x, Set.Ioo_subset_Icc_self hx⟩
  have hdecay : SourceCoeffQuadraticDecay p (u t) :=
    sourceCoeffQuadraticDecay_of_solution hsol ht
  have hR : intervalNeumannResolverR p (u t) y =
      intervalDomainLift (v t) x := by
    simpa [y] using
      solution_v_eq_resolver_pointwise_unconditional
        (p := p) (T := T) (u := u) (v := v) hsol ht hx
  have hSrc : intervalNeumannResolverSourceValue p (u t) y =
      p.ν * (intervalDomainLift (u t) x) ^ p.γ := by

      ShenWork.IntervalResolverLaplacianBridge.sourceValue_eq_source
        (p := p) (T := T) (u := u) (v := v) hsol ht y

    intervalNeumannResolverRLap p (u t) y
        = p.μ * intervalNeumannResolverR p (u t) y -
            intervalNeumannResolverSourceValue p (u t) y :=
          ShenWork.IntervalResolverLaplacianBridge.intervalNeumannResolverRLap_elliptic_identity
            hdecay y

          p.ν * (intervalDomainLift (u t) x) ^ p.γ := by
          rw [hR, hSrc]
