    (hM : 0 ≤ M) (hq : 0 < q)

        (-p.χ) • (∫ r in (0 : ℝ)..q,
          wholeLineCauchyGradientBUCIntegrand p hM hq.le A q r -
            wholeLineCauchyGradientBUCIntegrand p hM hq.le W q r) +

    {Wave : ℝ → ℝ}
    (hWaveUnif : UniformContinuous Wave)
    (hWaveIcc : ∀ x, Wave x ∈ Set.Icc (0 : ℝ) M)
