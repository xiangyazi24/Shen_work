# Q3978 (cron1): proving `hR_int` with `integral_tsum_pairing`

I could not fetch `ShenWork/Paper2/IntervalTruncatedTestedSpectral.lean` from the refs exposed by the connector, but the theorem statement in the prompt is enough to give the intended proof.  The main point is that `ψ` should be an **oriented interval indicator**.  If the local proof already has `hab : a ≤ b`, the proof is short with `ψ = 1_{(a,b]}`.  If the theorem only assumes `ha : a ∈ Ioo 0 1` and `hb : b ∈ Ioo 0 1`, split on `a ≤ b`; in the reverse branch use `ψ = -1_{(b,a]}`.

Below is the drop-in code I would use.  The only nontrivial bookkeeping is converting the `intervalMeasure 1` integral and the `0..1` interval integral back to the oriented interval integral `a..b`.

private lemma Ioc_subset_Icc_zero_one_of_mem_Ioo_of_le
    {a b : ℝ} (ha : a ∈ Set.Ioo (0 : ℝ) 1)
    (hb : b ∈ Set.Ioo (0 : ℝ) 1) (hab : a ≤ b) :
    Set.Ioc a b ⊆ Set.Icc (0 : ℝ) 1 := by
  intro x hx
  exact ⟨(lt_trans ha.1 hx.1).le, (lt_of_le_of_lt hx.2 hb.2).le⟩

private lemma Ioc_subset_Ioc_zero_one_of_mem_Ioo_of_le

    (hb : b ∈ Set.Ioo (0 : ℝ) 1) (hab : a ≤ b) :

/-- Convert the `intervalMeasure 1` pairing with `1_(a,b]` to the standard
oriented interval integral, in the forward-orientation case. -/
private lemma intervalMeasure_integral_mul_indicator_Ioc_one_eq_intervalIntegral_of_le
    {f : ℝ → ℝ} {a b : ℝ}
    (ha : a ∈ Set.Ioo (0 : ℝ) 1)
    (hb : b ∈ Set.Ioo (0 : ℝ) 1) (hab : a ≤ b) :
    (∫ x, f x * Set.indicator (Set.Ioc a b) (fun _ : ℝ => (1 : ℝ)) x
        ∂ intervalMeasure 1)
      = ∫ x in a..b, f x := by
  have hprod :
      (fun x : ℝ => f x * Set.indicator (Set.Ioc a b)
        (fun _ : ℝ => (1 : ℝ)) x)
        = Set.indicator (Set.Ioc a b) f := by
    funext x
    by_cases hx : x ∈ Set.Ioc a b <;> simp [Set.indicator, hx]
  have hsub : Set.Ioc a b ⊆ Set.Icc (0 : ℝ) 1 :=
    Ioc_subset_Icc_zero_one_of_mem_Ioo_of_le ha hb hab
  calc
    (∫ x, f x * Set.indicator (Set.Ioc a b) (fun _ : ℝ => (1 : ℝ)) x
        ∂ intervalMeasure 1)
        = ∫ x, Set.indicator (Set.Ioc a b) f x ∂ intervalMeasure 1 := by
          rw [hprod]
    _ = ∫ x in Set.Ioc a b, f x ∂ intervalMeasure 1 := by
          rw [MeasureTheory.integral_indicator measurableSet_Ioc]
    _ = ∫ x in Set.Ioc a b, f x := by
          unfold intervalMeasure intervalSet
          change ∫ x, f x ∂ (volume.restrict (Set.Icc (0 : ℝ) 1)).restrict (Set.Ioc a b)
            = ∫ x, f x ∂ volume.restrict (Set.Ioc a b)
          rw [Measure.restrict_restrict measurableSet_Ioc]
          congr 1
          ext x
          by_cases hx : x ∈ Set.Ioc a b
          · simp [hx, hsub hx]
          · simp [hx]
    _ = ∫ x in a..b, f x := by
          rw [intervalIntegral.integral_of_le hab]

/-- Convert the `0..1` side of `integral_tsum_pairing` with `1_(a,b]`. -/
private lemma intervalIntegral_zero_one_mul_indicator_Ioc_one_eq_intervalIntegral_of_le
    {f : ℝ → ℝ} {a b : ℝ}
    (ha : a ∈ Set.Ioo (0 : ℝ) 1)
    (hb : b ∈ Set.Ioo (0 : ℝ) 1) (hab : a ≤ b) :
    (∫ x in (0 : ℝ)..1,
        f x * Set.indicator (Set.Ioc a b) (fun _ : ℝ => (1 : ℝ)) x)
      = ∫ x in a..b, f x := by
  have hprod :
      (fun x : ℝ => f x * Set.indicator (Set.Ioc a b)
        (fun _ : ℝ => (1 : ℝ)) x)
        = Set.indicator (Set.Ioc a b) f := by
    funext x
    by_cases hx : x ∈ Set.Ioc a b <;> simp [Set.indicator, hx]
  have hsub : Set.Ioc a b ⊆ Set.Ioc (0 : ℝ) 1 :=
    Ioc_subset_Ioc_zero_one_of_mem_Ioo_of_le ha hb hab
  calc
    (∫ x in (0 : ℝ)..1,
        f x * Set.indicator (Set.Ioc a b) (fun _ : ℝ => (1 : ℝ)) x)
        = ∫ x in Set.Ioc (0 : ℝ) 1,
            f x * Set.indicator (Set.Ioc a b) (fun _ : ℝ => (1 : ℝ)) x := by
          rw [intervalIntegral.integral_of_le (by norm_num : (0 : ℝ) ≤ 1)]
    _ = ∫ x in Set.Ioc (0 : ℝ) 1, Set.indicator (Set.Ioc a b) f x := by
          rw [hprod]
    _ = ∫ x in Set.Ioc a b, f x := by
          change ∫ x, Set.indicator (Set.Ioc a b) f x ∂ volume.restrict (Set.Ioc (0 : ℝ) 1)
            = ∫ x, f x ∂ volume.restrict (Set.Ioc a b)
          rw [MeasureTheory.integral_indicator measurableSet_Ioc]
          change ∫ x, f x ∂ (volume.restrict (Set.Ioc (0 : ℝ) 1)).restrict (Set.Ioc a b)
            = ∫ x, f x ∂ volume.restrict (Set.Ioc a b)
          rw [Measure.restrict_restrict measurableSet_Ioc]
          congr 1
          ext x
          by_cases hx : x ∈ Set.Ioc a b
          · simp [hx, hsub hx]
          · simp [hx]
    _ = ∫ x in a..b, f x := by
          rw [intervalIntegral.integral_of_le hab]

/-- Reverse-orientation version for the left side, using `-1_(b,a]`. -/
private lemma intervalMeasure_integral_mul_neg_indicator_Ioc_one_eq_intervalIntegral_of_ge
    {f : ℝ → ℝ} {a b : ℝ}
    (ha : a ∈ Set.Ioo (0 : ℝ) 1)
    (hb : b ∈ Set.Ioo (0 : ℝ) 1) (hba : b ≤ a) :
    (∫ x, f x * (fun y : ℝ =>
        -Set.indicator (Set.Ioc b a) (fun _ : ℝ => (1 : ℝ)) y) x
        ∂ intervalMeasure 1)
      = ∫ x in a..b, f x := by
  calc
    (∫ x, f x * (fun y : ℝ =>
        -Set.indicator (Set.Ioc b a) (fun _ : ℝ => (1 : ℝ)) y) x
        ∂ intervalMeasure 1)
        = - ∫ x, f x * Set.indicator (Set.Ioc b a) (fun _ : ℝ => (1 : ℝ)) x
            ∂ intervalMeasure 1 := by
          rw [← integral_neg]
          congr 1
          funext x
          ring
    _ = - ∫ x in b..a, f x := by
          rw [intervalMeasure_integral_mul_indicator_Ioc_one_eq_intervalIntegral_of_le
            (f := f) hb ha hba]
    _ = ∫ x in a..b, f x := by
          rw [intervalIntegral.integral_symm]
          ring

/-- Reverse-orientation version for the right `0..1` side, using `-1_(b,a]`. -/
private lemma intervalIntegral_zero_one_mul_neg_indicator_Ioc_one_eq_intervalIntegral_of_ge

    (ha : a ∈ Set.Ioo (0 : ℝ) 1)
    (hb : b ∈ Set.Ioo (0 : ℝ) 1) (hba : b ≤ a) :

## If the local theorem already has `a ≤ b`

If the surrounding proof has a hypothesis `hab : a ≤ b`, keep only the first branch.  The essential core is:

```lean
let ψ : ℝ → ℝ := Set.indicator (Set.Ioc a b) (fun _ : ℝ => (1 : ℝ))

have hpair := integral_tsum_pairing
  (w := resolverValueSeriesReal p u) (ψ := ψ)
  (c := C) (B := fun _ : ℕ => (1 : ℝ))
  (e := fun k x => unitIntervalCosineMode k x)
  (Cψ := (1 : ℝ))
  he heB hsum hrep hψm hψb

rw [intervalMeasure_integral_mul_indicator_Ioc_one_eq_intervalIntegral_of_le
      (f := resolverValueSeriesReal p u) ha hb hab] at hpair
simpa [J, modeInt, ψ] using hpair
```

The reason the full branch split is safer is that `∫ x in a..b, ...` is an oriented interval integral.  A plain nonnegative indicator `1_[a,b]` only represents it when `a ≤ b`; otherwise one must use `-1_(b,a]`.
