The existing `summable_cell_heat_interval_integral` gives summability of the **combined** nonnegative left+right heat cell masses.  The smallest faithful step is to first split that combined summability into the two nonnegative branch-mass summabilities, then use a bounded-measurable multiplier estimate plus `Summable.of_norm_bounded`.

The key existing pattern is already in `IntervalConjugateKernelMassDefect.lean`: `reflectedKernelPart_integral_summable` splits one nonnegative summand from `summable_cell_heat_interval_integral` using `Summable.of_nonneg_of_le`, `intervalIntegral.integral_nonneg_of_forall`, and `le_add_of_nonneg_left`.  The code below mirrors that pattern and adds the missing left-branch analogue plus the bounded multiplier theorem.

/-- A bounded measurable multiplier preserves an interval-integral bound against a
nonnegative interval-integrable kernel.  This is the reusable local atom for the
folded image branches. -/
private theorem interval_integral_abs_mul_le_of_nonneg_bound
    {g f : ℝ → ℝ} {M : ℝ}
    (hM : 0 ≤ M)
    (hf_meas : Measurable f) (hf_bound : ∀ y, |f y| ≤ M)
    (hg_nonneg : ∀ y, 0 ≤ g y)
    (hg_int : IntervalIntegrable g volume (0 : ℝ) 1) :
    |∫ y in (0 : ℝ)..1, g y * f y| ≤
      M * ∫ y in (0 : ℝ)..1, g y := by
  have h01 : (0 : ℝ) ≤ 1 := by norm_num
  have hg_restrict :
      Integrable g (volume.restrict (Set.Ioc (0 : ℝ) 1)) :=
    (intervalIntegrable_iff_integrableOn_Ioc_of_le h01).mp hg_int
  have hf_ae : AEStronglyMeasurable f (volume.restrict (Set.Ioc (0 : ℝ) 1)) :=
    hf_meas.aestronglyMeasurable
  have hprod_restrict :
      Integrable (fun y : ℝ => g y * f y)
        (volume.restrict (Set.Ioc (0 : ℝ) 1)) :=
    hg_restrict.mul_bdd hf_ae
      (Filter.Eventually.of_forall fun y => by
        simpa [Real.norm_eq_abs] using hf_bound y)
  have hprod_int :
      IntervalIntegrable (fun y : ℝ => g y * f y) volume (0 : ℝ) 1 :=
    (intervalIntegrable_iff_integrableOn_Ioc_of_le h01).mpr hprod_restrict
  have hdom_int :
      IntervalIntegrable (fun y : ℝ => M * g y) volume (0 : ℝ) 1 :=
    hg_int.const_mul M
  have hpoint : ∀ y ∈ Set.Ioo (0 : ℝ) 1, |g y * f y| ≤ M * g y := by
    intro y _hy
    calc
      |g y * f y| = g y * |f y| := by
        rw [abs_mul, abs_of_nonneg (hg_nonneg y)]
      _ ≤ g y * M := mul_le_mul_of_nonneg_left (hf_bound y) (hg_nonneg y)
      _ = M * g y := by ring
  calc
    |∫ y in (0 : ℝ)..1, g y * f y|
        ≤ ∫ y in (0 : ℝ)..1, |g y * f y| :=
          intervalIntegral.abs_integral_le_integral_abs h01
    _ ≤ ∫ y in (0 : ℝ)..1, M * g y :=
          intervalIntegral.integral_mono_on_of_le_Ioo h01 hprod_int.abs hdom_int hpoint
    _ = M * ∫ y in (0 : ℝ)..1, g y := by
          rw [intervalIntegral.integral_const_mul]

/-- Summability of the left image-branch cell masses. -/
theorem heatKernel_leftBranch_interval_integral_summable

    Summable (fun k : ℤ =>
      ∫ y in (0 : ℝ)..1, heatKernel t (x - y + 2 * (k : ℝ))) := by

/-- Bounded measurable data make both folded image-branch interval-integral
families summable. -/
theorem foldedHeat_branch_summable_of_bounded
    {t : ℝ} (ht : 0 < t) (x : intervalDomainPoint) {f : ℝ → ℝ} {M : ℝ}
    (hM : 0 ≤ M) (hf_meas : Measurable f) (hf_bound : ∀ y, |f y| ≤ M) :
    Summable (fun k : ℤ =>
      ∫ y in (0 : ℝ)..1, heatKernel t (-x.1 - y + 2 * (k : ℝ)) * f y) ∧
    Summable (fun k : ℤ =>
      ∫ y in (0 : ℝ)..1, heatKernel t (-x.1 + y + 2 * (k : ℝ)) * f y) := by
  have hhc : Continuous (fun w : ℝ => heatKernel t w) := by
    unfold heatKernel
    fun_prop
  constructor
  · have hheat : Summable (fun k : ℤ =>
        ∫ y in (0 : ℝ)..1, heatKernel t (-x.1 - y + 2 * (k : ℝ))) :=
      heatKernel_leftBranch_interval_integral_summable ht (-x.1)
    refine Summable.of_norm_bounded
      (g := fun k : ℤ => M *
        ∫ y in (0 : ℝ)..1, heatKernel t (-x.1 - y + 2 * (k : ℝ)))
      (hheat.mul_left M) ?_
    intro k
    rw [Real.norm_eq_abs]
    have haff : Continuous (fun y : ℝ => -x.1 - y + 2 * (k : ℝ)) := by
      fun_prop
    have hg_int :
        IntervalIntegrable
          (fun y : ℝ => heatKernel t (-x.1 - y + 2 * (k : ℝ)))
          volume (0 : ℝ) 1 :=
      (hhc.comp haff).intervalIntegrable 0 1
    exact interval_integral_abs_mul_le_of_nonneg_bound hM hf_meas hf_bound
      (fun y => heatKernel_nonneg ht _) hg_int
  · have hheat : Summable (fun k : ℤ =>
        ∫ y in (0 : ℝ)..1, heatKernel t (-x.1 + y + 2 * (k : ℝ))) :=
      heatKernel_rightBranch_interval_integral_summable ht (-x.1)
    refine Summable.of_norm_bounded
      (g := fun k : ℤ => M *
        ∫ y in (0 : ℝ)..1, heatKernel t (-x.1 + y + 2 * (k : ℝ)))
      (hheat.mul_left M) ?_
    intro k
    rw [Real.norm_eq_abs]
    have haff : Continuous (fun y : ℝ => -x.1 + y + 2 * (k : ℝ)) := by
      fun_prop
    have hg_int :
        IntervalIntegrable
          (fun y : ℝ => heatKernel t (-x.1 + y + 2 * (k : ℝ)))
          volume (0 : ℝ) 1 :=
      (hhc.comp haff).intervalIntegrable 0 1
    exact interval_integral_abs_mul_le_of_nonneg_bound hM hf_meas hf_bound
      (fun y => heatKernel_nonneg ht _) hg_int

end
