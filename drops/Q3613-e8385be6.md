```lean

```

It does **not** use endpoint compatibility, mass-defect control, endpoint operator vanish, or any future Dirichlet positivity theorem.  It is a pure oscillation estimate:

```lean
|∫ (df y - df x) K̃(t,x,y)|
  ≤ ∫ |df y - df x| |K̃(t,x,y)|
  ≤ ε/2 ∫ |K̃| + C ∫ |y-x| |K̃|
  ≤ ε/2 + C · O(√t).
```

Constants:

* Uniform-continuity tolerance: `ε / 2`.
* Global bound: `|df| ≤ M`, with `M > 0`.
* Linear modulus coefficient: `C := 2 * M / ρ`, where `ρ` is the uniform-continuity radius.
* Time horizon: `δ := (ε / (4 * C)) ^ 2`.
* Existing square-root estimate:

```lean
4 * t / Real.sqrt (4 * Real.pi * t) ≤ 2 * Real.sqrt t
```

then gives `C * moment < ε / 2` for `t < δ`.

```lean
/-- Continuous derivative profiles have vanishing conjugate-kernel oscillation term. -/
theorem initialLegConjugateOscillationControl_of_continuousOn
    {df : ℝ → ℝ}
    (hdf_cont : ContinuousOn df (Set.Icc (0 : ℝ) 1)) :
    InitialLegConjugateOscillationControl df := by
  intro ε hε
  -- A positive uniform sup bound for `df` on `[0,1]`.
  obtain ⟨M, hM_pos, hdfM⟩ :
      ∃ M : ℝ, 0 < M ∧ ∀ y ∈ Set.Icc (0 : ℝ) 1, |df y| ≤ M := by
    obtain ⟨B, hB⟩ := isCompact_Icc.exists_bound_of_continuousOn hdf_cont
    refine ⟨max B 1, by positivity, ?_⟩
    intro y hy
    exact (Real.norm_eq_abs (df y) ▸ hB y hy).trans (le_max_left B 1)

  -- Uniform continuity on the compact interval, with tolerance `ε/2`.
  have huc := isCompact_Icc.uniformContinuousOn_of_continuous
    (s := Set.Icc (0 : ℝ) 1) hdf_cont
  rw [Metric.uniformContinuousOn_iff] at huc
  obtain ⟨ρ, hρ_pos, hρdf⟩ := huc (ε / 2) (by linarith)

  -- Linear modulus: close points use uniform continuity; far points use the
  -- global bound and absorb `2M` into `C * |y-x|`.
  set C : ℝ := 2 * M / ρ with hC_def
  have hC_pos : 0 < C := by
    rw [hC_def]
    positivity
  have hlinmod : ∀ x ∈ Set.Icc (0 : ℝ) 1, ∀ y ∈ Set.Icc (0 : ℝ) 1,
      |df y - df x| ≤ ε / 2 + C * |y - x| := by
    intro x hx y hy
    by_cases hclose : dist y x < ρ
    · have hdist := hρdf y hy x hx hclose
      rw [Real.dist_eq] at hdist
      calc |df y - df x|
          ≤ ε / 2 := le_of_lt hdist
        _ ≤ ε / 2 + C * |y - x| := by
          exact le_add_of_nonneg_right (mul_nonneg hC_pos.le (abs_nonneg _))
    · have hdist_ge : ρ ≤ dist y x := le_of_not_gt hclose
      have hyx : ρ ≤ |y - x| := by
        simpa [Real.dist_eq] using hdist_ge
      have hdiff_le : |df y - df x| ≤ 2 * M := by
        have hfy := abs_le.mp (hdfM y hy)
        have hfx := abs_le.mp (hdfM x hx)
        exact abs_le.mpr ⟨by linarith, by linarith⟩
      have hlinear : 2 * M ≤ C * |y - x| := by
        calc
          2 * M = C * ρ := by
            rw [hC_def]
            field_simp [ne_of_gt hρ_pos]
            ring
          _ ≤ C * |y - x| :=
            mul_le_mul_of_nonneg_left hyx hC_pos.le
      calc |df y - df x|
          ≤ 2 * M := hdiff_le
        _ ≤ C * |y - x| := hlinear
        _ ≤ ε / 2 + C * |y - x| := by linarith

  -- Choose the same square-root horizon as in the full-kernel uniform AI proof.
  set δ : ℝ := (ε / (4 * C)) ^ 2 with hδ_def
  have hδ_pos : 0 < δ := by
    rw [hδ_def]
    positivity
  refine ⟨δ, hδ_pos, ?_⟩
  intro t ht htδ x hx

  let K : ℝ → ℝ := fun y => intervalNeumannConjugateKernel t x y
  have h01 : (0 : ℝ) ≤ 1 := by norm_num
  have hdf_u : ContinuousOn df (Set.uIcc (0 : ℝ) 1) := by
    simpa [Set.uIcc_of_le h01] using hdf_cont
  have hK_u : ContinuousOn K (Set.uIcc (0 : ℝ) 1) := by
    simpa [K, Set.uIcc_of_le h01] using continuousOn_conjugateKernel_snd ht x
  have hdist_u : ContinuousOn (fun y : ℝ => |y - x|) (Set.uIcc (0 : ℝ) 1) :=
    (continuous_abs.comp (continuous_id.sub continuous_const)).continuousOn

  -- Interval-integrability of all terms used below.
  have hprod_ii : IntervalIntegrable
      (fun y : ℝ => (df y - df x) * K y) MeasureTheory.volume 0 1 :=
    ((hdf_u.sub continuousOn_const).mul hK_u).intervalIntegrable
  have hprod_abs_ii : IntervalIntegrable
      (fun y : ℝ => |(df y - df x) * K y|) MeasureTheory.volume 0 1 :=
    hprod_ii.abs
  have hKabs_ii : IntervalIntegrable
      (fun y : ℝ => |K y|) MeasureTheory.volume 0 1 :=
    hK_u.abs.intervalIntegrable
  have hmoment_ii : IntervalIntegrable
      (fun y : ℝ => |y - x| * |K y|) MeasureTheory.volume 0 1 :=
    (hdist_u.mul hK_u.abs).intervalIntegrable
  have hmod_ii : IntervalIntegrable
      (fun y : ℝ => (ε / 2 + C * |y - x|) * |K y|)
      MeasureTheory.volume 0 1 := by
    have hmod_u : ContinuousOn
        (fun y : ℝ => (ε / 2 + C * |y - x|) * |K y|)
        (Set.uIcc (0 : ℝ) 1) :=
      ((continuousOn_const.add (continuousOn_const.mul hdist_u)).mul hK_u.abs)
    exact hmod_u.intervalIntegrable

  change |-(∫ y in (0 : ℝ)..1, (df y - df x) * K y)| < ε

  -- Main absolute-value estimate using interval-integral triangle/norm bound.
  have h_integral_bound :
      |-(∫ y in (0 : ℝ)..1, (df y - df x) * K y)| ≤
        ∫ y in (0 : ℝ)..1, (ε / 2 + C * |y - x|) * |K y| := by
    calc
      |-(∫ y in (0 : ℝ)..1, (df y - df x) * K y)|
          = |∫ y in (0 : ℝ)..1, (df y - df x) * K y| := by
            rw [abs_neg]
      _ ≤ ∫ y in (0 : ℝ)..1, |(df y - df x) * K y| :=
            intervalIntegral.abs_integral_le_integral_abs h01
      _ ≤ ∫ y in (0 : ℝ)..1, (ε / 2 + C * |y - x|) * |K y| := by
            refine intervalIntegral.integral_mono_on h01 hprod_abs_ii hmod_ii ?_
            intro y hy
            have hyIcc : y ∈ Set.Icc (0 : ℝ) 1 := by
              simpa [Set.uIcc_of_le h01] using hy
            rw [abs_mul]
            exact mul_le_mul_of_nonneg_right
              (hlinmod x hx y hyIcc) (abs_nonneg (K y))

  -- Split the modulus integral into an `L¹` piece and a first-moment piece.
  have hsplit :
      (∫ y in (0 : ℝ)..1, (ε / 2 + C * |y - x|) * |K y|)
        = (ε / 2) * (∫ y in (0 : ℝ)..1, |K y|)
          + C * (∫ y in (0 : ℝ)..1, |y - x| * |K y|) := by
    have hfun :
        (fun y : ℝ => (ε / 2 + C * |y - x|) * |K y|)
          = fun y : ℝ =>
              (ε / 2) * |K y| + C * (|y - x| * |K y|) := by
      funext y
      ring
    rw [hfun]
    rw [intervalIntegral.integral_add
      (hKabs_ii.const_mul (ε / 2)) (hmoment_ii.const_mul C)]
    rw [intervalIntegral.integral_const_mul, intervalIntegral.integral_const_mul]

  have hmod_bound :
      (∫ y in (0 : ℝ)..1, (ε / 2 + C * |y - x|) * |K y|)
        ≤ ε / 2 + C * (4 * t / Real.sqrt (4 * Real.pi * t)) := by
    rw [hsplit]
    have hmass : (∫ y in (0 : ℝ)..1, |K y|) ≤ 1 := by
      simpa [K] using conjugateKernel_L1_bound ht x
    have hmoment :
        (∫ y in (0 : ℝ)..1, |y - x| * |K y|)
          ≤ 4 * t / Real.sqrt (4 * Real.pi * t) := by
      simpa [K] using conjugateKernel_abs_moment_le ht hx
    have hmass_part :
        (ε / 2) * (∫ y in (0 : ℝ)..1, |K y|) ≤ (ε / 2) * 1 :=
      mul_le_mul_of_nonneg_left hmass (by linarith)
    have hmoment_part :
        C * (∫ y in (0 : ℝ)..1, |y - x| * |K y|)
          ≤ C * (4 * t / Real.sqrt (4 * Real.pi * t)) :=
      mul_le_mul_of_nonneg_left hmoment hC_pos.le
    calc
      (ε / 2) * (∫ y in (0 : ℝ)..1, |K y|)
          + C * (∫ y in (0 : ℝ)..1, |y - x| * |K y|)
          ≤ (ε / 2) * 1 + C * (4 * t / Real.sqrt (4 * Real.pi * t)) :=
            add_le_add hmass_part hmoment_part
      _ = ε / 2 + C * (4 * t / Real.sqrt (4 * Real.pi * t)) := by ring

  -- Same square-root tail estimate used by `intervalFullSemigroup_tendstoUniformlyOn`.
  have htail_bound : C * (4 * t / Real.sqrt (4 * Real.pi * t)) < ε / 2 := by
    have h4pit_pos : 0 < 4 * Real.pi * t := by positivity
    have hpi_ge : 4 * t ≤ 4 * Real.pi * t := by
      nlinarith [Real.pi_gt_three]
    have hsqrt4t : Real.sqrt (4 * t) = 2 * Real.sqrt t := by
      have h4t_eq : (4 : ℝ) * t = (2 * Real.sqrt t) * (2 * Real.sqrt t) := by
        have := Real.mul_self_sqrt ht.le
        nlinarith
      rw [show (4 : ℝ) * t = (2 * Real.sqrt t) * (2 * Real.sqrt t) from h4t_eq,
        Real.sqrt_mul_self (by positivity : (0 : ℝ) ≤ 2 * Real.sqrt t)]
    have hmoment_le : 4 * t / Real.sqrt (4 * Real.pi * t) ≤ 2 * Real.sqrt t := by
      rw [div_le_iff₀ (Real.sqrt_pos_of_pos h4pit_pos)]
      calc
        4 * t = 2 * Real.sqrt t * Real.sqrt (4 * t) := by
          rw [hsqrt4t]
          nlinarith [Real.mul_self_sqrt ht.le]
        _ ≤ 2 * Real.sqrt t * Real.sqrt (4 * Real.pi * t) :=
          mul_le_mul_of_nonneg_left (Real.sqrt_le_sqrt hpi_ge) (by positivity)
    have hsqrt_bound : Real.sqrt t < ε / (4 * C) := by
      rw [← Real.sqrt_sq (show (0 : ℝ) ≤ ε / (4 * C) by positivity)]
      rw [hδ_def] at htδ
      exact Real.sqrt_lt_sqrt ht.le htδ
    calc
      C * (4 * t / Real.sqrt (4 * Real.pi * t))
          ≤ C * (2 * Real.sqrt t) :=
            mul_le_mul_of_nonneg_left hmoment_le hC_pos.le
      _ < C * (2 * (ε / (4 * C))) :=
            mul_lt_mul_of_pos_left (by linarith) hC_pos
      _ = ε / 2 := by
            field_simp
            ring

  calc
    |-(∫ y in (0 : ℝ)..1, (df y - df x) * K y)|
        ≤ ∫ y in (0 : ℝ)..1, (ε / 2 + C * |y - x|) * |K y| :=
          h_integral_bound
    _ ≤ ε / 2 + C * (4 * t / Real.sqrt (4 * Real.pi * t)) :=
          hmod_bound
    _ < ε := by linarith

end

end ShenWork.IntervalSemigroupC1ApproxIdentity
```

## Notes on likely compile adjustments

1. If the local moment lemma was placed in namespace `ShenWork.IntervalNeumannFullKernel` rather than `ShenWork.IntervalSemigroupC1ApproxIdentity`, replace

```lean
conjugateKernel_abs_moment_le ht hx
```

with


or open that namespace.

2. The proof intentionally uses `IntervalIntegrable` for the two interval-integral comparison calls:

```lean
intervalIntegral.integral_mono_on h01 hprod_abs_ii hmod_ii ...
intervalIntegral.integral_add ...
```

This avoids needing a separate conversion to `intervalMeasure` and keeps it aligned with the existing `IntervalFullKernelSourceIBP.lean` style.

3. Full positivity of `-intervalNeumannConjugateKernel` is not involved.  The oscillation term only needs absolute-value bounds, and those are supplied by `conjugateKernel_L1_bound` and `conjugateKernel_abs_moment_le`.

4. Endpoint compatibility is not needed here.  It enters later through mass-defect and endpoint-layer operator vanish, not the oscillation control.
