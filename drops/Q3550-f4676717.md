## 1. Derivative of the logistic reaction part

Paste this first. It is the delicate real-rpow part; it uses `hsol.u_pos'` to justify the real-power chain rule.

```lean
/-- Fixed-time derivative of the logistic H¹ reaction factor.

The real-rpow chain rule uses strict positivity of the classical solution. -/
theorem H1PhysicalLogisticReactionPart_hasDerivAt_of_classicalSolution
    {p : CM2Params} {T τ x : ℝ}
    {u v : ℝ → intervalDomainPoint → ℝ}
    (hsol : IsPaper2ClassicalSolution intervalDomain p T u v)
    (hτ : τ ∈ Set.Ioo (0 : ℝ) T)
    (hx : x ∈ Set.Ioo (0 : ℝ) 1) :
    HasDerivAt
      (fun y => H1PhysicalLogisticReactionPart p u τ y)
      (H1PhysicalLogisticSlope p u τ x *
        deriv (intervalDomainLift (u τ)) x)
      x := by
  let U : ℝ → ℝ := fun y => intervalDomainLift (u τ) y
  have hxIcc : x ∈ Set.Icc (0 : ℝ) 1 := Set.Ioo_subset_Icc_self hx
  have hC2u : ContDiffOn ℝ 2 U (Set.Icc (0 : ℝ) 1) := by
    simpa [U] using (hsol.regularity.2.2.2.2.1 τ hτ).1.1
  have hUderiv : HasDerivAt U (deriv U x) x :=
    hasDerivAt_of_contDiffOn_two_interior hC2u hx
  have hUpos : 0 < U x := by
    have hpos : 0 < u τ (⟨x, hxIcc⟩ : intervalDomainPoint) :=
      hsol.u_pos' hτ.1 hτ.2
    simpa [U, intervalDomainLift, hxIcc] using hpos
  have hpow0 := hUderiv.rpow_const (p := p.α) (Or.inl (ne_of_gt hUpos))
  have hpow :
      HasDerivAt (fun y => U y ^ p.α)
        (deriv U x * p.α * U x ^ (p.α - 1)) x := by
    refine hpow0.congr_deriv ?_
    ring_nf
  have hfactor :
      HasDerivAt (fun y => p.a - p.b * U y ^ p.α)
        ((-p.b) * (deriv U x * p.α * U x ^ (p.α - 1))) x := by
    simpa [sub_eq_add_neg, mul_assoc] using (hpow.const_mul (-p.b)).const_add p.a
  have hprod := hUderiv.mul hfactor
  have hpow_mul : U x * U x ^ (p.α - 1) = U x ^ p.α := by
    calc
      U x * U x ^ (p.α - 1)
          = U x ^ (1 : ℝ) * U x ^ (p.α - 1) := by rw [Real.rpow_one]
      _ = U x ^ ((1 : ℝ) + (p.α - 1)) := by rw [Real.rpow_add hUpos]
      _ = U x ^ p.α := by ring_nf
  have hderiv_eq :
      deriv U x * (p.a - p.b * U x ^ p.α) +
          U x * ((-p.b) * (deriv U x * p.α * U x ^ (p.α - 1))) =
        H1PhysicalLogisticSlope p u τ x * deriv U x := by
    unfold H1PhysicalLogisticSlope
    dsimp [U] at hpow_mul ⊢
    rw [← hpow_mul]
    ring
  have hprod' := hprod.congr_deriv hderiv_eq
  simpa [H1PhysicalLogisticReactionPart, H1PhysicalLogisticSlope, U,
    mul_assoc, mul_left_comm, mul_comm] using hprod'
```

```lean
/-- Logistic reaction scalar after one spatial integration by parts. -/
theorem H1PhysicalReactX_eq_logisticSlope_grad_sq_of_classicalSolution
    {p : CM2Params} {T τ : ℝ}
    {u v : ℝ → intervalDomainPoint → ℝ}
    (hsol : IsPaper2ClassicalSolution intervalDomain p T u v)
    (hτ : τ ∈ Set.Ioo (0 : ℝ) T) :
    H1PhysicalReactX p u τ =
      ∫ x in (0 : ℝ)..1,
        H1PhysicalLogisticSlope p u τ x *
          (deriv (intervalDomainLift (u τ)) x) ^ 2 := by
  let U : ℝ → ℝ := fun x => intervalDomainLift (u τ) x
  let R : ℝ → ℝ := fun x => H1PhysicalLogisticReactionPart p u τ x
  let dU : ℝ → ℝ := fun x => deriv U x
  let ddU : ℝ → ℝ := fun x => deriv (fun y => deriv U y) x
  let S : ℝ → ℝ := fun x => H1PhysicalLogisticSlope p u τ x
  have hC2u : ContDiffOn ℝ 2 U (Set.Icc (0 : ℝ) 1) := by
    simpa [U] using (hsol.regularity.2.2.2.2.1 τ hτ).1.1
  have hdw0 : derivWithin U (Set.Icc (0 : ℝ) 1) 0 = 0 := by
    simpa [U] using
      intervalDomain_solution_derivWithin_u_left_zero
        (params := p) (T := T) (t := τ) (u := u) (v := v)
        hsol hτ.1 hτ.2
  have hdw1 : derivWithin U (Set.Icc (0 : ℝ) 1) 1 = 0 := by
    simpa [U] using
      intervalDomain_solution_derivWithin_u_right_zero
        (params := p) (T := T) (t := τ) (u := u) (v := v)
        hsol hτ.1 hτ.2
  have hR_cont : ContinuousOn R (Set.uIcc (0 : ℝ) 1) := by
    rw [Set.uIcc_of_le (by norm_num : (0 : ℝ) ≤ 1)]
    simpa [R] using
      H1PhysicalLogisticReactionPart_continuousOn_Icc_of_classicalSolution
        (p := p) (T := T) (τ := τ) (u := u) (v := v) hsol hτ
  have hdU_cont : ContinuousOn dU (Set.uIcc (0 : ℝ) 1) := by
    rw [Set.uIcc_of_le (by norm_num : (0 : ℝ) ≤ 1)]
    simpa [dU, U] using
      deriv_intervalDomainLift_continuousOn_Icc_of_regularity hC2u hdw0 hdw1
  have hR_deriv : ∀ x ∈ Set.Ioo (min (0 : ℝ) 1) (max (0 : ℝ) 1),
      HasDerivAt R (S x * dU x) x := by
    intro x hx
    have hx01 : x ∈ Set.Ioo (0 : ℝ) 1 := by
      simpa [min_eq_left (by norm_num : (0 : ℝ) ≤ 1),
        max_eq_right (by norm_num : (0 : ℝ) ≤ 1)] using hx
    simpa [R, S, dU, U] using
      H1PhysicalLogisticReactionPart_hasDerivAt_of_classicalSolution
        (p := p) (T := T) (τ := τ) (u := u) (v := v)
        hsol hτ hx01
  have hdU_deriv : ∀ x ∈ Set.Ioo (min (0 : ℝ) 1) (max (0 : ℝ) 1),
      HasDerivAt dU (ddU x) x := by
    intro x hx
    have hx01 : x ∈ Set.Ioo (0 : ℝ) 1 := by
      simpa [min_eq_left (by norm_num : (0 : ℝ) ≤ 1),
        max_eq_right (by norm_num : (0 : ℝ) ≤ 1)] using hx
    simpa [dU, ddU, U] using
      hasDerivAt_deriv_of_contDiffOn_two_interior hC2u hx01
  have hS_cont : ContinuousOn S (Set.uIcc (0 : ℝ) 1) := by
    rw [Set.uIcc_of_le (by norm_num : (0 : ℝ) ≤ 1)]
    simpa [S] using
      H1PhysicalLogisticSlope_continuousOn_Icc_of_classicalSolution
        (p := p) (T := T) (τ := τ) (u := u) (v := v) hsol hτ
  have hR_deriv_int : IntervalIntegrable (fun x => S x * dU x) volume 0 1 :=
    (hS_cont.mul hdU_cont).intervalIntegrable
  have hddU_int : IntervalIntegrable ddU volume 0 1 := by
    simpa [ddU, U, liftDeriv2] using
      intervalIntegrable_deriv_deriv_of_contDiffOn_two hC2u
  have hraw :
      (∫ x in (0 : ℝ)..1, R x * ddU x) =

  have hdU0 : dU 0 = 0 := by
    dsimp [dU, U]
    exact deriv_intervalDomainLift_eq_zero_at_zero (u τ)
  have hdU1 : dU 1 = 0 := by
    dsimp [dU, U]
    exact deriv_intervalDomainLift_eq_zero_at_one (u τ)

      (∫ x in (0 : ℝ)..1, R x * ddU x) =
        -∫ x in (0 : ℝ)..1, (S x * dU x) * dU x := by
    rw [hraw, hdU0, hdU1]
    ring
  have hcomm :
      (∫ x in (0 : ℝ)..1, ddU x * R x) =
        ∫ x in (0 : ℝ)..1, R x * ddU x := by
    apply intervalIntegral.integral_congr
    intro x hx
    ring
  have hsq :
      (∫ x in (0 : ℝ)..1, (S x * dU x) * dU x) =
        ∫ x in (0 : ℝ)..1, S x * (dU x) ^ 2 := by
    apply intervalIntegral.integral_congr
    intro x hx
    ring
  calc
    H1PhysicalReactX p u τ
        = -(∫ x in (0 : ℝ)..1, ddU x * R x) := by
            simp [H1PhysicalReactX, R, ddU, U, liftDeriv2]
    _ = -(∫ x in (0 : ℝ)..1, R x * ddU x) := by rw [hcomm]
    _ = ∫ x in (0 : ℝ)..1, (S x * dU x) * dU x := by rw [hraw_zero]; ring
    _ = ∫ x in (0 : ℝ)..1, S x * (dU x) ^ 2 := hsq
    _ = ∫ x in (0 : ℝ)..1,
          H1PhysicalLogisticSlope p u τ x *
            (deriv (intervalDomainLift (u τ)) x) ^ 2 := by
          simp [S, dU, U]
```

5. The rpow algebra must keep the positivity proof at `U x`. The crucial identity is:

```lean
U x * U x ^ (p.α - 1) = U x ^ p.α
```

proved with `Real.rpow_add hUpos`, not by `ring` alone.

This closes only the reaction part of the physical sqrt frontier. It does not discharge taxis/uvxx L² estimates, zero-start primitive production, or any headline bounded-before theorem.
