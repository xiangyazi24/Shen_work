## Core theorem 1: normalized IBP with integrable/representative source

First factor out the real dependency reduction in the divergence-mode identity. The current

```lean
cosineCoeffs_deriv_eq_sqrtLambda_sineCoeff_open
```

requires `Continuous Q'`, but the representative route only needs interval integrability of the derivative source. Prove:

```lean
import ShenWork.Paper2.IntervalSourceBridgeOpen

open MeasureTheory intervalIntegral
open scoped Real

noncomputable section

namespace ShenWork.Paper2.IntervalSourceBridgeOpen

open ShenWork.IntervalNeumannFullKernel (cosineCoeffs)
open ShenWork.Paper2.HSigmaScale (lam)
open ShenWork.Paper2.IntervalDivergenceModeIdentity
  (sineCoeffs sineCoeffs_zero sineCoeffs_pos sqrt_lam_eq_kpi)

/-- Normalized open/right-derivative IBP with only interval-integrability of the
source derivative.  This is the endpoint-safe version needed for a continuous
representative of the chem-div source. -/
theorem cosineCoeffs_deriv_eq_sqrtLambda_sineCoeff_open_integrable
    {Q Q' : ℝ → ℝ} (k : ℕ)
    (hQcont : ContinuousOn Q (Set.Icc (0 : ℝ) 1))
    (hQderiv : ∀ x ∈ Set.Ioo (0 : ℝ) 1,
      HasDerivWithinAt Q (Q' x) (Set.Ioi x) x)
    (hQ'int : IntervalIntegrable Q' volume (0 : ℝ) 1)
    (hQ0 : Q 0 = 0) (hQ1 : Q 1 = 0) :
    cosineCoeffs Q' k = Real.sqrt (lam k) * sineCoeffs Q k := by
  rcases Nat.eq_zero_or_pos k with rfl | hk
  · rw [ShenWork.IntervalMildPicardRegularity.cosineCoeffs_zero_eq_integral,
        sineCoeffs_zero, mul_zero]
    have h01 : (0 : ℝ) ≤ 1 := by norm_num
    have hint : (∫ x in (0 : ℝ)..1, Q' x) = Q 1 - Q 0 := by
      exact intervalIntegral.integral_eq_sub_of_hasDeriv_right_of_le
        h01 hQcont hQderiv hQ'int
    rw [hint, hQ0, hQ1, sub_zero]
  · have hkne : k ≠ 0 := Nat.pos_iff_ne_zero.mp hk
    rw [ShenWork.IntervalMildPicardRegularity.cosineCoeffs_pos_eq_integral hkne,
        sineCoeffs_pos hkne, sqrt_lam_eq_kpi]
    have hraw := rawCosCoeff_deriv_eq_kpi_rawSinCoeff_open k
      hQcont hQderiv hQ'int hQ0 hQ1
    have hcomm :
        (∫ x in (0 : ℝ)..1, Real.cos ((k : ℝ) * Real.pi * x) * Q' x)
          = ∫ x in (0 : ℝ)..1, Q' x * Real.cos ((k : ℝ) * Real.pi * x) := by
      refine intervalIntegral.integral_congr (fun x _ => ?_); ring
    have hsincomm :
        (∫ x in (0 : ℝ)..1, Q x * Real.sin ((k : ℝ) * Real.pi * x))
          = ∫ x in (0 : ℝ)..1, Real.sin ((k : ℝ) * Real.pi * x) * Q x := by
      refine intervalIntegral.integral_congr (fun x _ => ?_); ring
    rw [hcomm, hraw, hsincomm]
    ring

end ShenWork.Paper2.IntervalSourceBridgeOpen
```

```lean
/-- Divergence-mode identity from an endpoint-insensitive continuous representative

theorem divMode_of_sliceC1_open_representative
    {p : CM2Params} {u : ℝ → intervalDomainPoint → ℝ} {s : ℝ}
    (hQcont : ContinuousOn (chemFluxLifted p (u s)) (Set.Icc (0 : ℝ) 1))
    (hQderiv : ∀ x ∈ Set.Ioo (0 : ℝ) 1,
      HasDerivWithinAt (chemFluxLifted p (u s))
        (coupledChemDivSourceLift p u s x) (Set.Ioi x) x)
    {Gdiv : ℝ → ℝ}
    (hGcont : ContinuousOn Gdiv (Set.Icc (0 : ℝ) 1))
    (hGeq : Set.EqOn (coupledChemDivSourceLift p u s) Gdiv (Set.Ioo (0 : ℝ) 1))
    (n : ℕ) :
    ((n : ℝ) * Real.pi) * intervalSineInner (chemFluxLifted p (u s)) n
      = coupledChemDivSourceCoeffs p u s n := by
  have hQ0 : chemFluxLifted p (u s) 0 = 0 := chemFluxLifted_endpoint_zero p (u s)
  have hQ1 : chemFluxLifted p (u s) 1 = 0 := chemFluxLifted_endpoint_one p (u s)
  have hGint : IntervalIntegrable Gdiv volume (0 : ℝ) 1 := by
    -- `hGcont.intervalIntegrable_of_Icc (by norm_num)` is the local expected shape.
    exact hGcont.intervalIntegrable_of_Icc (by norm_num : (0 : ℝ) ≤ 1)
  have hQderivG : ∀ x ∈ Set.Ioo (0 : ℝ) 1,
      HasDerivWithinAt (chemFluxLifted p (u s)) (Gdiv x) (Set.Ioi x) x := by
    intro x hx
    have h := hQderiv x hx
    rw [hGeq hx] at h
    exact h
  have hibp := cosineCoeffs_deriv_eq_sqrtLambda_sineCoeff_open_integrable
    (Q := chemFluxLifted p (u s)) (Q' := Gdiv) n
    hQcont hQderivG hGint hQ0 hQ1
  have hcoeff : cosineCoeffs (coupledChemDivSourceLift p u s) n = cosineCoeffs Gdiv n :=
    ShenWork.Paper2.BankChemSliceFix.cosineCoeffs_congr_on_Ioo
      (fun y hy => hGeq hy) n
  rw [coupledChemDivSourceCoeffs, hcoeff, hibp,
    sqrt_lam_eq_kpi, sineInner_eq_sineCoeffs]

end ShenWork.Paper2.IntervalSourceBridgeOpen
```

```lean
* logistic source continuity is required on the subtype source profile, not on the

    (hlog_bound : ∀ n, |cosineCoeffs (logisticLifted p (u s)) n| ≤ Mlog)
    {Mchem : ℝ}
    (hchem_bound : ∀ n, |coupledChemDivSourceCoeffs p u s n| ≤ Mchem)
    (hQderiv : ∀ y ∈ Set.Ioo (0 : ℝ) 1,

        (coupledChemDivSourceLift p u s y) (Set.Ioi y) y)
```
