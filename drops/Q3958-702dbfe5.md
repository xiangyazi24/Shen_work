Implemented `gradDuhamel_shifted_sup_bound` by shifting the source path from `[a,t]` to `[0,t-a]` and applying the existing unshifted `ShenWork.IntervalGradDuhamelBound.gradDuhamel_sup_bound`.

The key reduction is:

```lean
let F : ℝ → ℝ := fun s : ℝ =>
  deriv (fun z : ℝ => intervalFullSemigroupOperator (t - s) (q s) z) x
let qshift : ℝ → ℝ → ℝ := fun r : ℝ => q (r + a)
```

Then the shifted integrand is pointwise the translated original integrand:

```lean
(fun r : ℝ => deriv
  (fun z : ℝ => intervalFullSemigroupOperator ((t - a) - r) (qshift r) z) x)
  = fun r : ℝ => F (r + a)
```

The proof transports `hg_int` with `IntervalIntegrable.comp_add_right`, calls the existing unshifted bound with `t := t - a` and `T := T - a`, and rewrites the shifted interval integral using `intervalIntegral.integral_comp_add_right`.

    {q : ℝ → ℝ → ℝ}
    (hq_int : ∀ s, Integrable (q s) (intervalMeasure 1))
    {Cq : ℝ} (hCq : 0 ≤ Cq) (hq_sup : ∀ s y, |q s y| ≤ Cq) (x : ℝ)

      ≤ heatGradientLinftyLinftyConstant * (2 * Real.sqrt (T - a)) * Cq := by
  let F : ℝ → ℝ := fun s : ℝ =>
    deriv (fun z : ℝ => intervalFullSemigroupOperator (t - s) (q s) z) x
  let qshift : ℝ → ℝ → ℝ := fun r : ℝ => q (r + a)
  have hdt_pos : 0 < t - a := sub_pos.mpr hat
  have hdtT : t - a ≤ T - a := sub_le_sub_right htT a
  have hq_int_shift : ∀ r, Integrable (qshift r) (intervalMeasure 1) := by
    intro r
    dsimp [qshift]
    exact hq_int (r + a)
  have hq_sup_shift : ∀ r y, |qshift r y| ≤ Cq := by
    intro r y
    dsimp [qshift]
    exact hq_sup (r + a) y
  have hgF_int : IntervalIntegrable F volume a t := by
    simpa [F] using hg_int
  have hF_shift :
      (fun r : ℝ => deriv
        (fun z : ℝ => intervalFullSemigroupOperator ((t - a) - r) (qshift r) z) x)
        = fun r : ℝ => F (r + a) := by
    funext r
    dsimp [F, qshift]
    rw [show (t - a) - r = t - (r + a) by ring]
  have hg_shift : IntervalIntegrable
      (fun r : ℝ => deriv
        (fun z : ℝ => intervalFullSemigroupOperator ((t - a) - r) (qshift r) z) x)
      volume 0 (t - a) := by
    rw [hF_shift]
    simpa using (hgF_int.comp_add_right a)

      (t := t - a) (T := T - a) hdt_pos hdtT
      (q := qshift) hq_int_shift hCq hq_sup_shift x hg_shift
  have hIntegral_shift :
      (∫ r in (0 : ℝ)..(t - a), deriv
        (fun z : ℝ => intervalFullSemigroupOperator ((t - a) - r) (qshift r) z) x)
        = ∫ s in a..t, F s := by
    rw [hF_shift]
    have hcomp := intervalIntegral.integral_comp_add_right
      (f := F) (a := (0 : ℝ)) (b := t - a) a
    simpa using hcomp

      |∫ s in a..t, F s|
        ≤ heatGradientLinftyLinftyConstant * (2 * Real.sqrt (T - a)) * Cq := by
    rw [← hIntegral_shift]
    exact hbound_shift
  simpa [F] using hfinal
