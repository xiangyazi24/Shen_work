def PerDatumSpectralFrontierNoHvposNoInitialApproach
    (p : CM2Params) {u₀ : intervalDomainPoint → ℝ}
    (D : GradientMildSolutionData p u₀) : Prop :=
  ∃ _S : GradientMildHalfStepLogisticSourceData D,
  HasTimeNeighborhoodSpectralAgreement D.T D.u ∧
  HasResolverDirectSpectralData D.T

    intervalDomain.timeDeriv D.u t x =
      intervalDomain.laplacian (D.u t) x

            (mildChemicalConcentration p D.u t) x
        + D.u t x * (p.a - p.b * (D.u t x) ^ p.α))

/-- Fill the deleted `hVpos` and `hInitialApproach` fields from already proved
infrastructure, yielding the existing `PerDatumSpectralFrontier` interface. -/
theorem perDatumSpectralFrontier_of_noHvpos_noInitialApproach
    (p : CM2Params) {u₀ : intervalDomainPoint → ℝ}
    (hu₀ : PositiveInitialDatum intervalDomain u₀)
    (D : GradientMildSolutionData p u₀)
    (H : PerDatumSpectralFrontierNoHvposNoInitialApproach p D) :

    (p : CM2Params)
    (hPerDatum : ∀ u₀ : intervalDomainPoint → ℝ,

## Proof route

* `hMildLocal_of_perDatum_noHvpos_noInitialApproach` is a one-line composition with existing `EndToEnd.hMildLocal_of_perDatum`.

This should typecheck with the imports above. The only possible minor syntax adjustment is namespace qualification for `IntervalDomainGradientMildHalfStepLogisticSourceFrontierCoreLocalData`; if Lean does not find it unqualified, use the same imports/opens as `IntervalDomainEndToEnd.lean`, or qualify it from the theorem source module.
