# Q3928 — sine coefficient IBP bound on `[0,1]`

## The Mathlib/API choice

The right API for this proof in this repository is not a specialized `integral_mul_deriv_of_le` lemma.  Use the interval-integral FTC lemma that the repo already uses in `IntervalDuhamelSourceTimeC1On.lean`:

```lean
intervalIntegral.integral_eq_sub_of_hasDeriv_right_of_le
```

The clean proof is: define

```lean
ω := (k : ℝ) * Real.pi
A x := -Real.cos (ω * x) / ω
```

so `A' x = Real.sin (ω * x)`.  Apply the FTC lemma to the product

```lean
F x := A x * Q x
```

whose derivative is

```lean
Real.sin (ω*x) * Q x + A x * deriv Q x.
```

Then solve for `∫ sin(ωx) Q x`, multiply by `2*ω`, and bound the two boundary terms and the derivative integral.  This avoids fighting a more specialized integration-by-parts theorem.

/-- Derivative of the antiderivative used for the sine IBP. -/
private theorem hasDerivAt_neg_cos_div_freq
    {ω x : ℝ} (hω : ω ≠ 0) :
    HasDerivAt (fun y : ℝ => -Real.cos (ω * y) / ω)
      (Real.sin (ω * x)) x := by
  have hlin : HasDerivAt (fun y : ℝ => ω * y) ω x := by
    simpa using (hasDerivAt_id x).const_mul ω
  have hcos : HasDerivAt (fun y : ℝ => Real.cos (ω * y))
      (-(Real.sin (ω * x) * ω)) x := by
    simpa [mul_comm, mul_left_comm, mul_assoc] using
      (Real.hasDerivAt_cos (ω * x)).comp x hlin
  have h := hcos.neg.div_const ω
  convert h using 1
  · ext y
    ring
  · field_simp [hω]

/-- `|∫ cos(ωx) Q'(x)| ≤ ∫ |Q'(x)|`.  This is a tiny wrapper around the usual
interval-integral norm bound. -/
private theorem abs_integral_cos_mul_deriv_le
    {Q : ℝ → ℝ} {ω : ℝ}
    (hQ'_int : IntegrableOn (deriv Q) (Icc (0 : ℝ) 1) volume) :
    |∫ x in (0 : ℝ)..1, Real.cos (ω * x) * deriv Q x|
      ≤ ∫ x in (0 : ℝ)..1, |deriv Q x| := by
  have hbound : ∀ x ∈ Ι (0 : ℝ) 1,
      ‖Real.cos (ω * x) * deriv Q x‖ ≤ |deriv Q x| := by
    intro x hx
    rw [Real.norm_eq_abs, abs_mul]
    exact mul_le_of_le_one_left (abs_nonneg _) (Real.abs_cos_le_one _)
  -- If this exact line does not elaborate, replace it by the project-local
  -- intervalIntegral norm lemma used in `IntervalBFormSpectralPdeAgreementStandardFacts`:
  -- `intervalIntegral.norm_integral_le_of_norm_le_const` is for constant bounds;
  -- the variable-bound version is the standard `norm_integral_le_integral_norm`.
  have h1 :
      ‖∫ x in (0 : ℝ)..1, Real.cos (ω * x) * deriv Q x‖
        ≤ ∫ x in (0 : ℝ)..1, ‖Real.cos (ω * x) * deriv Q x‖ := by
    exact intervalIntegral.norm_integral_le_integral_norm _
  have h2 :
      (∫ x in (0 : ℝ)..1, ‖Real.cos (ω * x) * deriv Q x‖)
        ≤ ∫ x in (0 : ℝ)..1, |deriv Q x| := by
    refine intervalIntegral.integral_mono_on (by norm_num : (0 : ℝ) ≤ 1) ?_ ?_ ?_
    · exact hQ'_int.norm.intervalIntegrable
    · exact (hQ'_int.mul_continuousOn
        ((continuous_cos.comp (continuous_const.mul continuous_id)).continuousOn)).norm.intervalIntegrable
    · intro x hx
      simpa [Real.norm_eq_abs] using hbound x hx
  simpa [Real.norm_eq_abs] using h1.trans h2

/-- The exact IBP identity behind the estimate. -/
private theorem freq_mul_intervalSineInner_eq_boundary_plus_deriv
    {Q : ℝ → ℝ} {k : ℕ}
    (hk : k ≠ 0)
    (hQ_cont : ContinuousOn Q (Icc (0 : ℝ) 1))
    (hQ_deriv : ∀ x ∈ Ioo (0 : ℝ) 1,
      HasDerivAt Q (deriv Q x) x)
    (hQ'_integrable :
      IntegrableOn (deriv Q) (Icc (0 : ℝ) 1) volume) :
    ((k : ℝ) * Real.pi) * intervalSineInner Q k =
      2 * (-Real.cos ((k : ℝ) * Real.pi) * Q 1 + Q 0)
        + 2 * ∫ x in (0 : ℝ)..1,
          Real.cos ((k : ℝ) * Real.pi * x) * deriv Q x := by
  classical
  set ω : ℝ := (k : ℝ) * Real.pi with hωdef
  have hkpos_nat : 0 < k := Nat.pos_of_ne_zero hk
  have hω_ne : ω ≠ 0 := by
    rw [hωdef]
    exact mul_ne_zero (by exact_mod_cast (Nat.ne_of_gt hkpos_nat)) Real.pi_ne_zero
  let A : ℝ → ℝ := fun x => -Real.cos (ω * x) / ω
  have hA_deriv : ∀ x, HasDerivAt A (Real.sin (ω * x)) x := by
    intro x
    simpa [A] using hasDerivAt_neg_cos_div_freq (ω := ω) (x := x) hω_ne
  have hF_cont : ContinuousOn (fun x => A x * Q x) (Icc (0 : ℝ) 1) := by
    exact ((by fun_prop : Continuous A).continuousOn).mul hQ_cont
  have hF_deriv : ∀ x ∈ Ioo (0 : ℝ) 1,
      HasDerivWithinAt (fun y => A y * Q y)
        (Real.sin (ω * x) * Q x + A x * deriv Q x) (Ioi x) x := by
    intro x hx
    exact ((hA_deriv x).mul (hQ_deriv x hx)).hasDerivWithinAt
  have hA_derivQ_int : IntervalIntegrable (fun x => A x * deriv Q x) volume 0 1 := by
    -- bounded continuous multiplier times the carried `IntegrableOn` derivative
    -- This is usually one line in Mathlib/project code; adapt the method name if needed.
    exact (hQ'_integrable.mul_continuousOn
      ((by fun_prop : Continuous A).continuousOn)).intervalIntegrable
  have hsinQ_int : IntervalIntegrable (fun x => Real.sin (ω * x) * Q x) volume 0 1 := by
    apply ContinuousOn.intervalIntegrable
    rw [uIcc_of_le (by norm_num : (0 : ℝ) ≤ 1)]
    exact ((continuous_sin.comp (continuous_const.mul continuous_id)).continuousOn).mul hQ_cont
  have hderiv_int : IntervalIntegrable
      (fun x => Real.sin (ω * x) * Q x + A x * deriv Q x) volume 0 1 :=
    hsinQ_int.add hA_derivQ_int
  have hFTC := intervalIntegral.integral_eq_sub_of_hasDeriv_right_of_le
    (show (0 : ℝ) ≤ 1 by norm_num)
    (f := fun x => A x * Q x)
    (f' := fun x => Real.sin (ω * x) * Q x + A x * deriv Q x)
    hF_cont hF_deriv hderiv_int
  have hsplit :
      (∫ x in (0 : ℝ)..1,
          Real.sin (ω * x) * Q x + A x * deriv Q x)
        = (∫ x in (0 : ℝ)..1, Real.sin (ω * x) * Q x)
          + ∫ x in (0 : ℝ)..1, A x * deriv Q x := by
    exact intervalIntegral.integral_add hsinQ_int hA_derivQ_int
  rw [hsplit] at hFTC
  have hA0 : A 0 = -1 / ω := by simp [A]
  have hA1 : A 1 = -Real.cos ω / ω := by simp [A]
  have hI :
      (∫ x in (0 : ℝ)..1, Real.sin (ω * x) * Q x)
        = A 1 * Q 1 - A 0 * Q 0 -
          ∫ x in (0 : ℝ)..1, A x * deriv Q x := by
    linarith
  unfold intervalSineInner
  rw [if_neg hk]
  rw [hI]
  have hAint :
      (∫ x in (0 : ℝ)..1, A x * deriv Q x)
        = -(1 / ω) * ∫ x in (0 : ℝ)..1,
            Real.cos (ω * x) * deriv Q x := by
    unfold A
    rw [← intervalIntegral.integral_const_mul]
    refine intervalIntegral.integral_congr (fun x _ => ?_)
    ring
  rw [hA0, hA1, hAint]
  rw [show ((k : ℝ) * Real.pi) = ω by rw [hωdef]]
  field_simp [hω_ne]
  ring_nf

/-- Final frequency-weighted sine coefficient bound from a `W^{1,1}` slice. -/
theorem freq_mul_intervalSineInner_bound_of_W1
    {Q : ℝ → ℝ} {CQ Cder : ℝ}
    (hCQ : 0 ≤ CQ) (hCder : 0 ≤ Cder)
    (hQ_cont : ContinuousOn Q (Icc (0 : ℝ) 1))
    (hQ0 : |Q 0| ≤ CQ)
    (hQ1 : |Q 1| ≤ CQ)
    (hQ_deriv : ∀ x ∈ Ioo (0 : ℝ) 1,
      HasDerivAt Q (deriv Q x) x)
    (hQ'_integrable :
      IntegrableOn (deriv Q) (Icc (0 : ℝ) 1) volume)
    (hQ'_bound :
      (∫ x in (0 : ℝ)..1, |deriv Q x|) ≤ Cder) :
    ∀ k : ℕ,
      |((k : ℝ) * Real.pi) * intervalSineInner Q k| ≤
        4 * CQ + 2 * Cder := by
  intro k
  by_cases hk : k = 0
  · subst k
    simp [intervalSineInner, hCQ, hCder]
  · have hEq := freq_mul_intervalSineInner_eq_boundary_plus_deriv
      (Q := Q) (k := k) hk hQ_cont hQ_deriv hQ'_integrable
    rw [hEq]
    have hcos : |Real.cos ((k : ℝ) * Real.pi)| ≤ 1 := Real.abs_cos_le_one _
    have hboundary :
        |2 * (-Real.cos ((k : ℝ) * Real.pi) * Q 1 + Q 0)| ≤ 4 * CQ := by
      calc
        |2 * (-Real.cos ((k : ℝ) * Real.pi) * Q 1 + Q 0)|
            ≤ 2 * |Real.cos ((k : ℝ) * Real.pi)| * |Q 1| + 2 * |Q 0| := by
              -- This line is just `abs_add_le` plus `abs_mul`/`abs_neg`; `ring_nf` cleans constants.
              nlinarith [abs_add_le (2 * (-Real.cos ((k : ℝ) * Real.pi) * Q 1)) (2 * Q 0),
                abs_mul (2 : ℝ) (-Real.cos ((k : ℝ) * Real.pi) * Q 1),
                abs_mul (-Real.cos ((k : ℝ) * Real.pi)) (Q 1), abs_neg (Real.cos ((k : ℝ) * Real.pi))]
        _ ≤ 2 * 1 * CQ + 2 * CQ := by
              nlinarith [hcos, hQ0, hQ1, hCQ]
        _ = 4 * CQ := by ring
    have hint :
        |2 * ∫ x in (0 : ℝ)..1,
            Real.cos ((k : ℝ) * Real.pi * x) * deriv Q x|
          ≤ 2 * Cder := by
      calc
        |2 * ∫ x in (0 : ℝ)..1,
            Real.cos ((k : ℝ) * Real.pi * x) * deriv Q x|
            = 2 * |∫ x in (0 : ℝ)..1,
              Real.cos ((k : ℝ) * Real.pi * x) * deriv Q x| := by simp
        _ ≤ 2 * (∫ x in (0 : ℝ)..1, |deriv Q x|) := by
              exact mul_le_mul_of_nonneg_left
                (abs_integral_cos_mul_deriv_le (Q := Q)
                  (ω := (k : ℝ) * Real.pi) hQ'_integrable)
                (by norm_num)
        _ ≤ 2 * Cder := by nlinarith [hQ'_bound]
    calc
      |2 * (-Real.cos ((k : ℝ) * Real.pi) * Q 1 + Q 0)
          + 2 * ∫ x in (0 : ℝ)..1,
            Real.cos ((k : ℝ) * Real.pi * x) * deriv Q x|
          ≤ |2 * (-Real.cos ((k : ℝ) * Real.pi) * Q 1 + Q 0)|
              + |2 * ∫ x in (0 : ℝ)..1,
                Real.cos ((k : ℝ) * Real.pi * x) * deriv Q x| := abs_add_le _ _
      _ ≤ 4 * CQ + 2 * Cder := add_le_add hboundary hint

end ShenWork.Paper2
